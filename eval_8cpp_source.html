37int Evaluation::evaluate() {
38 const int phase = calculatePhase();
39 const int modifier = board.getSideToMove() == WHITE ? 1 : -1;
40
41 initializeEvalData();
42
43 evaluateMaterial();
44 evaluatePieces();
45
46 const int whiteScore = ((whiteMidgameScore * (256 - phase)) + (whiteEndgameScore * phase)) / 256;
47 const int blackScore = ((blackMidgameScore * (256 - phase)) + (blackEndgameScore * phase)) / 256;
48
49 return (whiteScore - blackScore) * modifier;
50}
51
52constexpr int knightPhase = 1;
53constexpr int bishopPhase = 1;
54constexpr int rookPhase = 2;
55constexpr int queenPhase = 4;
56constexpr int totalPhase = knightPhase * 4 + bishopPhase * 4 + rookPhase * 4 + queenPhase * 2;
57
67int Evaluation::calculatePhase() const {
68 int phase = totalPhase;
69
70 phase -= (board.getPieceCount<WHITE_KNIGHT>() + board.getPieceCount<BLACK_KNIGHT>()) * knightPhase;
71 phase -= (board.getPieceCount<WHITE_BISHOP>() + board.getPieceCount<BLACK_BISHOP>()) * bishopPhase;
72 phase -= (board.getPieceCount<WHITE_ROOK>() + board.getPieceCount<BLACK_ROOK>()) * rookPhase;
73 phase -= (board.getPieceCount<WHITE_QUEEN>() + board.getPieceCount<BLACK_QUEEN>()) * queenPhase;
74
75 return (phase * 256 + (totalPhase / 2)) / totalPhase;
76}

126void Evaluation::evaluatePieces() {
127 // Mobility
128 uint64_t whiteKnightMobility = evalData.attacksByPiece[WHITE_KNIGHT];
129 uint64_t blackKnightMobility = evalData.attacksByPiece[BLACK_KNIGHT];
130 uint64_t whiteBishopMobility = evalData.attacksByPiece[WHITE_BISHOP];
131 uint64_t blackBishopMobility = evalData.attacksByPiece[BLACK_BISHOP];
132 uint64_t whiteRookMobility = evalData.attacksByPiece[WHITE_ROOK];
133 uint64_t blackRookMobility = evalData.attacksByPiece[BLACK_ROOK];
134 uint64_t whiteQueenMobility = evalData.attacksByPiece[WHITE_QUEEN];
135 uint64_t blackQueenMobility = evalData.attacksByPiece[BLACK_QUEEN];
136
137 int whiteKnightMobilityCount = popcnt(whiteKnightMobility);
138 int blackKnightMobilityCount = popcnt(blackKnightMobility);
139 int whiteBishopMobilityCount = popcnt(whiteBishopMobility);
140 int blackBishopMobilityCount = popcnt(blackBishopMobility);
141 int whiteRookMobilityCount = popcnt(whiteRookMobility);
142 int blackRookMobilityCount = popcnt(blackRookMobility);
143 int whiteQueenMobilityCount = popcnt(whiteQueenMobility);
144 int blackQueenMobilityCount = popcnt(blackQueenMobility);
145
146 whiteMidgameScore += whiteKnightMobilityCount * MOBILITY_MIDGAME_KNIGHT_VALUE;
147 blackMidgameScore += blackKnightMobilityCount * MOBILITY_MIDGAME_KNIGHT_VALUE;
148 whiteMidgameScore += whiteBishopMobilityCount * MOBILITY_MIDGAME_BISHOP_VALUE;
149 blackMidgameScore += blackBishopMobilityCount * MOBILITY_MIDGAME_BISHOP_VALUE;
150 whiteMidgameScore += whiteRookMobilityCount * MOBILITY_MIDGAME_ROOK_VALUE;
151 blackMidgameScore += blackRookMobilityCount * MOBILITY_MIDGAME_ROOK_VALUE;
152 whiteMidgameScore += whiteQueenMobilityCount * MOBILITY_MIDGAME_QUEEN_VALUE;
153 blackMidgameScore += blackQueenMobilityCount * MOBILITY_MIDGAME_QUEEN_VALUE;
154
155 whiteEndgameScore += whiteKnightMobilityCount * MOBILITY_ENDGAME_KNIGHT_VALUE;
156 blackEndgameScore += blackKnightMobilityCount * MOBILITY_ENDGAME_KNIGHT_VALUE;
157 whiteEndgameScore += whiteBishopMobilityCount * MOBILITY_ENDGAME_BISHOP_VALUE;
158 blackEndgameScore += blackBishopMobilityCount * MOBILITY_ENDGAME_BISHOP_VALUE;
159 whiteEndgameScore += whiteRookMobilityCount * MOBILITY_ENDGAME_ROOK_VALUE;
160 blackEndgameScore += blackRookMobilityCount * MOBILITY_ENDGAME_ROOK_VALUE;
161 whiteEndgameScore += whiteQueenMobilityCount * MOBILITY_ENDGAME_QUEEN_VALUE;
162 blackEndgameScore += blackQueenMobilityCount * MOBILITY_ENDGAME_QUEEN_VALUE;
163}

168void Evaluation::initializeEvalData() {
169 const uint64_t occupiedBB = board.getOccupiedBitboard();
170 uint64_t pieces = occupiedBB;
171 const uint64_t whitePieces = board.getColorBitboard<WHITE>();
172 const uint64_t blackPieces = board.getColorBitboard<BLACK>();
173
174 while (pieces) {
175 const uint64_t pieceBB = popLsb(pieces);
176 const Square square = bitboardToSquare(pieceBB);
177 const Piece piece = board.getPieceOnSquare(square);
178
179 switch (piece) {
180 case WHITE_PAWN:
181 evalData.attacksFrom[square] = getPawnAttacks<WHITE>(square) & ~whitePieces;
182 evalData.attacksByColor[WHITE] |= evalData.attacksFrom[square];
183 evalData.attacksByPiece[WHITE_PAWN] |= evalData.attacksFrom[square];
184 break;
185 case BLACK_PAWN:
186 evalData.attacksFrom[square] = getPawnAttacks<BLACK>(square) & ~blackPieces;
187 evalData.attacksByColor[BLACK] |= evalData.attacksFrom[square];
188 evalData.attacksByPiece[BLACK_PAWN] |= evalData.attacksFrom[square];
189 break;
190 case WHITE_KNIGHT:
191 evalData.attacksFrom[square] = getKnightAttacks(square) & ~whitePieces;
192 evalData.attacksByColor[WHITE] |= evalData.attacksFrom[square];
193 evalData.attacksByPiece[WHITE_KNIGHT] |= evalData.attacksFrom[square];
194 break;
195 case BLACK_KNIGHT:
196 evalData.attacksFrom[square] = getKnightAttacks(square) & ~blackPieces;
197 evalData.attacksByColor[BLACK] |= evalData.attacksFrom[square];
198 evalData.attacksByPiece[BLACK_KNIGHT] |= evalData.attacksFrom[square];
199 break;
200 case WHITE_BISHOP:
201 evalData.attacksFrom[square] = getBishopAttacks(square, occupiedBB) & ~whitePieces;
202 evalData.attacksByColor[WHITE] |= evalData.attacksFrom[square];
203 evalData.attacksByPiece[WHITE_BISHOP] |= evalData.attacksFrom[square];
204 break;
205 case BLACK_BISHOP:
206 evalData.attacksFrom[square] = getBishopAttacks(square, occupiedBB) & ~blackPieces;
207 evalData.attacksByColor[BLACK] |= evalData.attacksFrom[square];
208 evalData.attacksByPiece[BLACK_BISHOP] |= evalData.attacksFrom[square];
209 break;
210 case WHITE_ROOK:
211 evalData.attacksFrom[square] = getRookAttacks(square, occupiedBB) & ~whitePieces;
212 evalData.attacksByColor[WHITE] |= evalData.attacksFrom[square];
213 evalData.attacksByPiece[WHITE_ROOK] |= evalData.attacksFrom[square];
214 break;
215 case BLACK_ROOK:
216 evalData.attacksFrom[square] = getRookAttacks(square, occupiedBB) & ~blackPieces;
217 evalData.attacksByColor[BLACK] |= evalData.attacksFrom[square];
218 evalData.attacksByPiece[BLACK_ROOK] |= evalData.attacksFrom[square];
219 break;
220 case WHITE_QUEEN:
221 evalData.attacksFrom[square] = queenAttacks(square, occupiedBB) & ~whitePieces;
222 evalData.attacksByColor[WHITE] |= evalData.attacksFrom[square];
223 evalData.attacksByPiece[WHITE_QUEEN] |= evalData.attacksFrom[square];
224 break;
225 case BLACK_QUEEN:
226 evalData.attacksFrom[square] = queenAttacks(square, occupiedBB) & ~blackPieces;
227 evalData.attacksByColor[BLACK] |= evalData.attacksFrom[square];
228 evalData.attacksByPiece[BLACK_QUEEN] |= evalData.attacksFrom[square];
229 break;
230 case WHITE_KING:
231 evalData.attacksFrom[square] = getKingAttacks(square) & ~whitePieces;
232 evalData.attacksByColor[WHITE] |= evalData.attacksFrom[square];
233 evalData.attacksByPiece[WHITE_KING] |= evalData.attacksFrom[square];
234 break;
235 case BLACK_KING:
236 evalData.attacksFrom[square] = getKingAttacks(square) & ~blackPieces;
237 evalData.attacksByColor[BLACK] |= evalData.attacksFrom[square];
238 evalData.attacksByPiece[BLACK_KING] |= evalData.attacksFrom[square];
239 break;
240 default:
241 break;
242 }
243 }
