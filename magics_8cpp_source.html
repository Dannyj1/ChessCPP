69int pop_1st_bit(uint64_t* bb) {
70 uint64_t b = *bb ^ (*bb - 1);
71 unsigned int fold = static_cast<unsigned>((b & 0xffffffff) ^ (b >> 32));
72 *bb &= (*bb - 1);
73 return BitTable[(fold * 0x783a9b23) >> 26];
74}

76uint64_t index_to_uint64_t(int index, int bits, uint64_t m) {
77 int i, j;
78 uint64_t result = 0ULL;
79 for (i = 0; i < bits; i++) {
80 j = pop_1st_bit(&m);
81 if (index & (1 << i)) result |= (1ULL << j);
82 }
83 return result;
84}

86uint64_t rmask(int sq) {
87 uint64_t result = 0ULL;
88 int rk = sq / 8, fl = sq % 8, r, f;
89 for (r = rk + 1; r <= 6; r++) result |= (1ULL << (fl + r * 8));
90 for (r = rk - 1; r >= 1; r--) result |= (1ULL << (fl + r * 8));
91 for (f = fl + 1; f <= 6; f++) result |= (1ULL << (f + rk * 8));
92 for (f = fl - 1; f >= 1; f--) result |= (1ULL << (f + rk * 8));
93 return result;
94}

150int transform(uint64_t b, uint64_t magic, int bits) {
151#if defined(USE_32_BIT_MULTIPLICATIONS)
152 return (unsigned)((int)b * (int)magic ^ (int)(b >> 32) * (int)(magic >> 32)) >> (32 - bits);
153#else
154 return static_cast<int>((b * magic) >> (64 - bits));
155#endif
156}

158uint64_t find_magic(int sq, int m, int bishop) {
159 uint64_t mask, b[4096], a[4096], used[4096], magic;
160 int i, j, k, n, fail;
161
162 mask = bishop ? bmask(sq) : rmask(sq);
163 n = count_1s(mask);
164
165 for (i = 0; i < (1 << n); i++) {
166 b[i] = index_to_uint64_t(i, n, mask);
167 a[i] = bishop ? batt(sq, b[i]) : ratt(sq, b[i]);
168 }
169 for (k = 0; k < 100000000; k++) {
170 magic = random_uint64_t_fewbits();
171 if (count_1s((mask * magic) & 0xFF00000000000000ULL) < 6) continue;
172 for (i = 0; i < 4096; i++) used[i] = 0ULL;
173 for (i = 0, fail = 0; !fail && i < (1 << n); i++) {
174 j = transform(b[i], magic, m);
175 if (used[j] == 0ULL)
176 used[j] = a[i];
177 else if (used[j] != a[i])
178 fail = 1;
179 }
180 if (!fail) return magic;
181 }
182 printf("***Failed***\n");
183 return 0ULL;
184}

186int count_bits(uint64_t bitboard) {
187 // bit size
188 int count = 0;
189
190 // pop bits untill bitboard is empty
191 while (bitboard) {
192 // increment size
193 count++;
194
195 // consecutively reset least significant 1st bit
196 bitboard &= bitboard - 1;
197 }
198
199 // return bit size
200 return count;
201}

204int get_ls1b_index(uint64_t bitboard) {
205 // make sure bitboard is not empty
206 if (bitboard != 0)
207 // convert trailing zeros before LS1B to ones and size them
208 return count_bits((bitboard & -bitboard) - 1);
209
210 // otherwise
211 // return illegal index
212 return -1;

215uint64_t set_occupancy(int index, int bits_in_mask, uint64_t attack_mask) {
216 // occupancy map
217 uint64_t occupancy = 0ULL;
218
219 // loop over the range of bits within attack mask
220 for (int count = 0; count < bits_in_mask; count++) {
221 // get LS1B index of attacks mask
222 int8_t square = get_ls1b_index(attack_mask);
223
224 // pop LS1B in attack map
225 pop_bit(attack_mask, square);
226
227 // make sure occupancy is on board
228 if (index & (1 << count))
229 // populate occupancy map
230 occupancy |= (1ULL << square);
231 }
232
233 // return occupancy map
234 return occupancy;
235}

353void init_sliders_attacks(int is_bishop) {
354 // loop over 64 board squares
355 for (int8_t square = 0; square < 64; square++) {
356 // init bishop & rook masks
357 bishop_masks[square] = mask_bishop_attacks(square);
358 rook_masks[square] = mask_rook_attacks(square);
359
360 // init current mask
361 uint64_t mask = is_bishop ? mask_bishop_attacks(square) : mask_rook_attacks(square);
362
363 // size attack mask bits
364 int bit_count = count_bits(mask);
365
366 // occupancy variations size
367 int occupancy_variations = 1 << bit_count;
368
369 // loop over occupancy variations
370 for (int count = 0; count < occupancy_variations; count++) {
371 // bishop
372 if (is_bishop) {
373 // init occupancies, magic index & attacks
374 uint64_t occupancy = set_occupancy(count, bit_count, mask);
375 uint64_t magic_index = occupancy * bishopMagics[square] >> (64 - BBits[square]);
376 bishop_attacks[square][magic_index] = bishop_attacks_on_the_fly(square, occupancy);
377 }
378
379 // rook
380 else {
381 // init occupancies, magic index & attacks
382 uint64_t occupancy = set_occupancy(count, bit_count, mask);
383 uint64_t magic_index = occupancy * rookMagics[square] >> (64 - RBits[square]);
384 rook_attacks[square][magic_index] = rook_attacks_on_the_fly(square, occupancy);
385 }
386 }
387 }

417void findFastestSeed() {
418 // variable with max double
419 uint64_t max = std::numeric_limits<uint64_t>::max();
420 std::random_device seedRd;
421 std::mt19937_64 seedGen(seedRd());
422 std::uniform_int_distribution<uint64_t> seedDis;
423
424 while (true) {
425 uint64_t seed = seedDis(seedGen);
426 uint64_t total = 0;
427
428 for (int i = 0; i < 50; i++) {
429 gen.seed(seedDis(seedGen));
430 std::chrono::time_point<std::chrono::steady_clock> start =
431 std::chrono::steady_clock::now();
432
433 generateMagics();
434
435 std::chrono::time_point<std::chrono::steady_clock> now =
436 std::chrono::steady_clock::now();
437 uint64_t elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(now - start).
438 count();
439 total += elapsed;
440 }
441
442 uint64_t average = total / 50;
443 if (average < max) {
444 max = average;
445 std::cout << "New fastest seed: 0x" << std::uppercase << std::hex << seed <<
446 std::nouppercase
447 << std::dec << "ULL, took: " << average << "ms (average)" << std::endl;
448 }
