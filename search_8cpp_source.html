41// TODO: Support more search variables (infinite, max nodes, etc.)
42template <PieceColor color>
43Move search(Engine& engine, Board& board, SearchParams& params, SearchStats& stats) {
44 constexpr PieceColor opponentColor = !color;
45
46 int depth = 1;
47 const int currentPly = board.getPly();
48 MoveList moves = MoveList{};
49 generateMoves<color, ALL>(board, moves);
50 MovePicker movePicker{moves};
51 Move bestMove = NO_MOVE;
52
53 int searchTime = calculateSearchTime<color>(params);
54 const auto endTime = std::chrono::steady_clock::now() + std::chrono::milliseconds(searchTime);
55 const auto startTime = std::chrono::steady_clock::now();
56
57 while (!engine.isSearchStopped() && (currentPly + depth) < MAX_PLY) {
58 if (params.blackTime > 0 || params.whiteTime > 0) {
59 // Don't start the next iteration if we are 10% away from the end time
60 if (depth > 1 && std::chrono::steady_clock::now() + std::chrono::milliseconds(searchTime / 10) > endTime) {
61 break;
62 }
63 }
64
65 if (params.depth > 0 && depth > params.depth) {
66 break;
67 }
68
69 movePicker.reset();
70 Move move;
71 int bestScore = std::numeric_limits<int>::min();
72 Move bestIterationMove{};
73
74 while (movePicker.next(move)) {
75 board.makeMove(movePicker.getCurrentMove());
76
77 if (!board.isPositionLegal<color>()) {
78 board.unmakeMove();
79 continue;
80 }
81
82 const int score = -pvSearch<opponentColor, ROOT>(board, INITIAL_ALPHA, INITIAL_BETA, depth, stats, endTime);
83
84 board.unmakeMove();
85
86 if (std::chrono::steady_clock::now() > endTime) {
87 break;
88 }
89
90 if (score > bestScore) {
91 bestScore = score;
92 bestIterationMove = move;
93 }
94 }
95
96 if (depth > 1 && std::chrono::steady_clock::now() > endTime) {
97 break;
98 }
99
100 bestMove = bestIterationMove;
101
102 // TODO: Implement PV
103 stats.timeSpentMs = std::chrono::duration_cast<std::chrono::milliseconds>(
104 std::chrono::steady_clock::now() - startTime).count();
105 stats.depth = depth;
106 stats.score = bestScore;
107
108 // Make sure we don't divide by zero
109 if (stats.timeSpentMs == 0) {
110 stats.timeSpentMs = 1;
111 }
112
113 uint64_t totalNodesSearch = stats.nodesSearched + stats.qNodesSearched;
114 uint64_t nps = static_cast<double>(totalNodesSearch) / (
115 static_cast<double>(stats.timeSpentMs) / 1000.0);
116 engine.sendInfoMessage(std::format("depth {} score cp {} nodes {} time {} nps {}", stats.depth,
117 stats.score, totalNodesSearch, stats.timeSpentMs, nps));
118 depth += 1;
119 }
120
121 if (bestMove == NO_MOVE) {
122 // Find the first legal move and play that
123 movePicker.reset();
124 Move move;
125
126 while (movePicker.next(move)) {
127 board.makeMove(movePicker.getCurrentMove());
128
129 if (!board.isPositionLegal<color>()) {
130 board.unmakeMove();
131 continue;
132 }
133
134 bestMove = move;
135 board.unmakeMove();
136 break;
137 }
138 }
139
140 return bestMove;
141}

146template <PieceColor color, NodeType nodeType>
147int pvSearch(Board& board, int alpha, int beta, int depth, SearchStats& stats,
148 const std::chrono::time_point<std::chrono::steady_clock>& endTime) {
149 if (board.isDraw()) {
150 return DRAW_SCORE;
151 }
152
153 if ((stats.nodesSearched + stats.qNodesSearched) % 1024 == 0 && std::chrono::steady_clock::now() > endTime) {
154 return beta;
155 }
156
157 if (depth == 0) {
158 return qSearch<color, nodeType>(board, alpha, beta, depth, stats, endTime);
159 }
160
161 stats.nodesSearched += 1;
162
163 constexpr bool isPV = nodeType == PV || nodeType == ROOT;
164 constexpr bool isRoot = nodeType == ROOT;
165 constexpr PieceColor opponentColor = !color;
166
167 if (!isPV) {
168 const int16_t score = tt->getScore(board.getZobristHash(), depth, alpha, beta, board.getPly());
169
170 if (score != NO_TT_SCORE) {
171 return score;
172 }
173 }
174
175 int bestScore = std::numeric_limits<int>::min();
176 bool firstMove = true;
177 int legalMoves = 0;
178
179 Move move;
180 MoveList moves = MoveList{};
181 generateMoves<color, ALL>(board, moves);
182 MovePicker movePicker{moves};
183 Move bestMove = NO_MOVE;
184
185 while (movePicker.next(move)) {
186 int score;
187 board.makeMove(move);
188
189 if (!board.isPositionLegal<color>()) {
190 board.unmakeMove();
191 continue;
192 }
193
194 legalMoves += 1;
195
196 if (firstMove) {
197 if (isRoot) {
198 score = -pvSearch<opponentColor, PV>(board, -beta, -alpha, depth - 1, stats, endTime);
199 } else {
200 score = -pvSearch<opponentColor, nodeType>(board, -beta, -alpha, depth - 1, stats, endTime);
201 }
202
203 firstMove = false;
204 } else {
205 score = -pvSearch<opponentColor, REGULAR>(board, -alpha - 1, -alpha, depth - 1, stats, endTime);
206
207 if (score > alpha && isPV) {
208 score = -pvSearch<opponentColor, PV>(board, -beta, -alpha, depth - 1, stats, endTime);
209 }
210 }
211
212 board.unmakeMove();
213
214 if (score >= beta) {
215 tt->storePosition(board.getZobristHash(), depth, board.getPly(), bestScore, bestMove, FAIL_HIGH_NODE);
216 return score;
217 }
218
219 if (score > bestScore) {
220 bestScore = score;
221 bestMove = move;
222
223 if (score > alpha) {
224 alpha = score;
225 }
226 }
227 }
228
229 if (!legalMoves) {
230 bool inCheck = board.isKingInCheck<color>();
231
232 if (inCheck) {
233 alpha = -MATE_SCORE + board.getPly();
234 } else {
235 alpha = DRAW_SCORE;
236 }
237 }
238
239 if (!isRoot) {
240 TTNodeType ttNodeType = FAIL_LOW_NODE;
241
242 if (isPV) {
243 ttNodeType = EXACT_NODE;
244 }
245
246 tt->storePosition(board.getZobristHash(), depth, board.getPly(), bestScore, bestMove, ttNodeType);
247 }
248
249 return alpha;
250}

252template <PieceColor color, NodeType nodeType>
253int qSearch(Board& board, int alpha, int beta, int depth, SearchStats& stats,
254 const std::chrono::time_point<std::chrono::steady_clock>
255 & endTime) {
256 constexpr bool isPV = nodeType == PV;
257
258 if (board.isDraw()) {
259 return DRAW_SCORE;
260 }
261
262 if (!isPV) {
263 const int16_t score = tt->getScore(board.getZobristHash(), depth, alpha, beta, board.getPly());
264
265 if (score != NO_TT_SCORE) {
266 return score;
267 }
268 }
269
270 if ((stats.nodesSearched + stats.qNodesSearched) % 1024 == 0 && std::chrono::steady_clock::now() > endTime) {
271 return beta;
272 }
273
274 stats.qNodesSearched += 1;
275
276 int bestScore = Evaluation(board).evaluate();
277
278 if (bestScore >= beta) {
279 tt->storePosition(board.getZobristHash(), depth, board.getPly(), bestScore, NO_MOVE, FAIL_HIGH_NODE);
280 return bestScore;
281 }
282
283 if (bestScore > alpha) {
284 alpha = bestScore;
285 }
286
287 Move move;
288 MoveList moves = MoveList{};
289 generateMoves<color, QSEARCH>(board, moves);
290 MovePicker movePicker{moves};
291
292 while (movePicker.next(move)) {
293 board.makeMove(move);
294
295 if (!board.isPositionLegal<color>()) {
296 board.unmakeMove();
297 continue;
298 }
299
300 const int score = -qSearch<!color, nodeType>(board, -beta, -alpha, depth - 1, stats, endTime);
301
302 board.unmakeMove();
303
304 if (score >= beta) {
305 tt->storePosition(board.getZobristHash(), depth, board.getPly(), bestScore, NO_MOVE, FAIL_HIGH_NODE);
306 return score;
307 }
308
309 if (score > bestScore) {
310 bestScore = score;
311 }
312
313 if (score > alpha) {
314 alpha = score;
315 }
316 }
317
318 TTNodeType ttNodeType = FAIL_LOW_NODE;
319
320 if (isPV) {
321 ttNodeType = EXACT_NODE;
322 }
323
324 tt->storePosition(board.getZobristHash(), depth, board.getPly(), bestScore, NO_MOVE, ttNodeType);
325
326 return bestScore;
327}
