35template <PieceColor color>
36void Evaluation::addScore(const int midgameScore, const int endgameScore) {
37 if (color == WHITE) {
38 whiteMidgameScore += midgameScore;
39 whiteEndgameScore += endgameScore;
40 } else {
41 blackMidgameScore += midgameScore;
42 blackEndgameScore += endgameScore;
43 }
44}

55int Evaluation::evaluate() {
56 const int phase = calculatePhase();
57 const int modifier = board.getSideToMove() == WHITE ? 1 : -1;
58
59 initializeEvalData();
60
61 evaluatePieces();
62
63 const int whiteScore = ((whiteMidgameScore * (256 - phase)) + (whiteEndgameScore * phase)) / 256;
64 const int blackScore = ((blackMidgameScore * (256 - phase)) + (blackEndgameScore * phase)) / 256;
65
66 return (whiteScore - blackScore) * modifier;
67}
68
69constexpr int knightPhase = 1;
70constexpr int bishopPhase = 1;
71constexpr int rookPhase = 2;
72constexpr int queenPhase = 4;
73constexpr int totalPhase = knightPhase * 4 + bishopPhase * 4 + rookPhase * 4 + queenPhase * 2;
74
84int Evaluation::calculatePhase() const {
85 int phase = totalPhase;
86
87 phase -= popcnt(board.getPieceBoard<WHITE_KNIGHT>() | board.getPieceBoard<BLACK_KNIGHT>()) * knightPhase;
88 phase -= popcnt(board.getPieceBoard<WHITE_BISHOP>() | board.getPieceBoard<BLACK_BISHOP>()) * bishopPhase;
89 phase -= popcnt(board.getPieceBoard<WHITE_ROOK>() | board.getPieceBoard<BLACK_ROOK>()) * rookPhase;
90 phase -= popcnt(board.getPieceBoard<WHITE_QUEEN>() | board.getPieceBoard<BLACK_QUEEN>()) * queenPhase;
91
92 return (phase * 256 + (totalPhase / 2)) / totalPhase;
93}

115template <PieceColor color>
116void Evaluation::evaluatePawns() {
117 constexpr Piece pawnPiece = color == WHITE ? WHITE_PAWN : BLACK_PAWN;
118 uint64_t pawns = board.getPieceBoard<pawnPiece>();
119
120 while (pawns) {
121 const Square square = static_cast<Square>(popLsb(pawns));
122 const int midgamePst = getMidgamePstEvalFeature(pawnPiece, square);
123 const int endgamePst = getEndgamePstEvalFeature(pawnPiece, square);
124
125 addScore<color>(midgamePst, endgamePst);
126
127 const uint64_t attacks = getPawnAttacks<color>(square);
128
129 evalData.attacksFrom[square] = attacks;
130 evalData.attacksByColor[color] |= attacks;
131 evalData.attacksByPiece[pawnPiece] |= attacks;
132 }
133}

277int getPieceValue(const Piece piece) {
278 switch (piece) {
279 case WHITE_PAWN:
280 case BLACK_PAWN:
281 return 100;
282 case WHITE_KNIGHT:
283 case BLACK_KNIGHT:
284 return 350;
285 case WHITE_BISHOP:
286 case BLACK_BISHOP:
287 return 350;
288 case WHITE_ROOK:
289 case BLACK_ROOK:
290 return 525;
291 case WHITE_QUEEN:
292 case BLACK_QUEEN:
293 return 1000;
294 case WHITE_KING:
295 case BLACK_KING:
296 return 0;
297 default:
298 assert(false);
299 return 0;
300 }
