26// Lookup tables for attack patterns
27static std::array<std::array<uint64_t, SQUARES>, COLORS> pawnAttacksTable{};
28static std::array<uint64_t, SQUARES> knightAttacksTable{};
29static std::array<uint64_t, SQUARES> kingAttacksTable{};
30
34void initializeAttackLookupTables() {
35 for (uint8_t square = 0; square < SQUARES; ++square) {
36 const uint64_t bb = squareToBitboard(square);
37
38 pawnAttacksTable[WHITE][square] = calculateWhitePawnAttacks(bb);
39 pawnAttacksTable[BLACK][square] = calculateBlackPawnAttacks(bb);
40 knightAttacksTable[square] = calculateKnightAttacks(bb);
41 kingAttacksTable[square] = calculateKingAttacks(bb);
42 }
43}

87uint64_t getBishopAttacks(const uint8_t square, uint64_t occupied) {
88 assert(square < SQUARES);
89 occupied &= getBishopMask(square);
90 occupied *= getBishopMagic(square);
91 occupied >>= 64 - BBits[square];
92
93 return getBishopMagicAttacks(square, occupied);
94}
95
102uint64_t getRookAttacks(const uint8_t square, uint64_t occupied) {
103 assert(square < SQUARES);
104 occupied &= getRookMask(square);
105 occupied *= getRookMagic(square);
106 occupied >>= 64 - RBits[square];
107
108 return getRookMagicAttacks(square, occupied);
109}

117uint64_t queenAttacks(const uint8_t square, const uint64_t occupied) {
118 assert(square < SQUARES);
119 return getBishopAttacks(square, occupied) | getRookAttacks(square, occupied);
120}
