234 template <Piece piece>
235 void setPiece(const uint8_t square) {
236 assert(piece != Piece::EMPTY);
237 assert(board[square] == Piece::EMPTY);
238 const uint64_t squareBB = squareToBitboard(square);
239
240 board[square] = piece;
241 bitboards[piece] |= squareBB;
242 occupied |= squareBB;
243 colorBoards[getPieceColor(piece)] |= squareBB;
244
245 const int zobristIndex = ZOBRIST_PIECE_START_INDEX + static_cast<int>(piece) * SQUARES + square;
246 zobristHash ^= getZobristConstant(zobristIndex);
247 }
