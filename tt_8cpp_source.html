30 const uint64_t index = zobristHash & hashSize;
31 TTEntry* entry = &transpositionTable[index];

50int16_t TranspositionTable::getScore(const uint64_t zobristHash, const int8_t depth, const int alpha, const int beta, const int ply) const {
51 const uint64_t index = zobristHash & hashSize;
52 const uint32_t validationHash = zobristHash >> 32;
53 const TTEntry* entry = &transpositionTable[index];
54
55 if (entry->validationHash == validationHash && entry->depth >= depth) {
56 bool returnScore = false;
57
58 if (entry->nodeType == EXACT_NODE) {
59 returnScore = true;
60 } else if (entry->nodeType == FAIL_LOW_NODE) {
61 if (entry->score <= alpha) {
62 returnScore = true;
63 }
64 } else if (entry->nodeType == FAIL_HIGH_NODE) {
65 if (entry->score >= beta) {
66 returnScore = true;
67 }
68 }
69
70 if (returnScore) {
71 int adjustedScore = entry->score;
72
73 if (adjustedScore >= MATE_SCORE) {
74 adjustedScore -= ply;
75 } else if (adjustedScore <= -MATE_SCORE) {
76 adjustedScore += ply;
77 }
78
79 return adjustedScore;
80 }
81 }
82
83 return INT16_MIN;
84}

92void TranspositionTable::setTableSize(int megaBytes) {
93 if ((megaBytes & (megaBytes - 1)) != 0) {
94 megaBytes = 1 << static_cast<int>(log2(megaBytes));
95 }
96
97 const uint64_t byteSize = megaBytes * 1024 * 1024;

99
100 delete[] transpositionTable;
101 transpositionTable = new TTEntry[entryCount]{};
102 hashSize = entryCount - 1;

104 for (uint64_t i = 0; i < entryCount; i++) {
