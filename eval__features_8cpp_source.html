50static std::vector<int> evalFeatures{};
51
52void initializeEvalFeatures() {
53 initializePst();
54
55 for (const int feature : baseEvalFeatures) {
56 evalFeatures.push_back(feature);
57 }
58
59 midgamePstStartIndex = evalFeatures.size();
60
61 for (Piece pieces = WHITE_PAWN; pieces <= BLACK_KING; pieces++) {
62 for (Square square = A1; square <= H8; square++) {
63 baseEvalFeatures.push_back(getMidgamePstBaseValue(pieces, square));
64 evalFeatures.push_back(getMidgamePstBaseValue(pieces, square));
65 }
66 }
67
68 endgamePstStartIndex = evalFeatures.size();
69
70 for (Piece pieces = WHITE_PAWN; pieces <= BLACK_KING; pieces++) {
71 for (Square square = A1; square <= H8; square++) {
72 baseEvalFeatures.push_back(getEndgamePstBaseValue(pieces, square));
73 evalFeatures.push_back(getEndgamePstBaseValue(pieces, square));
74 }
75 }
76}

78void updateFeatureValues(const std::vector<int>& values) {

82int getEvalFeatureValue(const EvalFeature feature) {

86int getMidgamePstEvalFeature(const Piece piece, const Square square) {

90int getEndgamePstEvalFeature(const Piece piece, const Square square) {
