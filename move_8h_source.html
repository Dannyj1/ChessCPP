67// bits 0-5: from square (0-63)
68// bits 6-11: to square (0-63)
69// bits 12-13: move type (00 = normal, 01 = promotion, 10 = en passant, 11 = castling)
70// bits 14-15: promotion piece (00 = queen, 01 = rook, 10 = bishop, 11 = knight)
71inline Move encodeMove(const uint8_t fromSquare, const uint8_t toSquare) {
72 // Assume normal move, so bits 12-13 are 00 and bits 14-15 are 00
73 return fromSquare | (toSquare << 6);
74}
75
76inline Move encodeMove(const uint8_t fromSquare, const uint8_t toSquare, const MoveType moveType) {
77 return fromSquare | (toSquare << 6) | (moveType << 12);
78}
79
80inline Move encodeMove(const uint8_t fromSquare, const uint8_t toSquare, const PromotionPiece promotionPiece) {
81 return fromSquare | (toSquare << 6) | (PROMOTION << 12) | (promotionPiece << 14);
82}

100inline Piece getPieceFromPromotionPiece(const PromotionPiece promotionPiece, const PieceColor color) {
101 switch (promotionPiece) {
102 case QUEEN_PROMOTION:
103 return static_cast<Piece>(WHITE_QUEEN + color);
104 case ROOK_PROMOTION:
105 return static_cast<Piece>(WHITE_ROOK + color);
106 case BISHOP_PROMOTION:
107 return static_cast<Piece>(WHITE_BISHOP + color);
108 case KNIGHT_PROMOTION:
109 return static_cast<Piece>(WHITE_KNIGHT + color);
110 default:
111 assert(false);
112 return EMPTY;
113 }
