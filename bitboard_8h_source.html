192template <Direction direction>
193constexpr uint64_t shift(const uint64_t bb) {
194 switch (direction) {
195 case NORTH:
196 return shiftNorth(bb);
197 case SOUTH:
198 return shiftSouth(bb);
199 case EAST:
200 return shiftEast(bb);
201 case WEST:
202 return shiftWest(bb);
203 case NORTH_EAST:
204 return shiftNorthEast(bb);
205 case NORTH_WEST:
206 return shiftNorthWest(bb);
207 case SOUTH_EAST:
208 return shiftSouthEast(bb);
209 case SOUTH_WEST:
210 return shiftSouthWest(bb);
211 case NORTH_NORTH_EAST:
212 return shiftNorthNorthEast(bb);
213 case NORTH_EAST_EAST:
214 return shiftNorthEastEast(bb);
215 case SOUTH_EAST_EAST:
216 return shiftSouthEastEast(bb);
217 case SOUTH_SOUTH_EAST:
218 return shiftSouthSouthEast(bb);
219 case SOUTH_SOUTH_WEST:
220 return shiftSouthSouthWest(bb);
221 case SOUTH_WEST_WEST:
222 return shiftSouthWestWest(bb);
223 default:
224 assert(false);
225 return bb;
226 }
227}

430inline uint64_t calculateKingAttacks(uint64_t bb) {
431 const uint64_t attacks = shiftEast(bb) | shiftWest(bb);
432 bb |= attacks;
433 return attacks | shiftNorth(bb) | shiftSouth(bb);
