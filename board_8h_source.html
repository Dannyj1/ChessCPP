193 template <Piece piece>
194 void setPiece(const uint8_t square) {
195 assert(piece != Piece::EMPTY);
196 assert(board[square] == Piece::EMPTY);
197 const uint64_t squareBB = squareToBitboard(square);
198
199 board[square] = piece;
200 bitboards[piece] |= squareBB;
201 occupied |= squareBB;
202 colorBoards[getPieceColor(piece)] |= squareBB;
203 pieceCounts[piece] += 1;
204 }
