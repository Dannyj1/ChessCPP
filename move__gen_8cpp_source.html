41template <PieceColor color, GenerationType type>
42void generateMoves(const Board& board, MoveList& moves) {
43 assert(moves.size == 0);
44
45 constexpr Piece ownKing = color == WHITE ? WHITE_KING : BLACK_KING;
46 constexpr Piece opponentKing = color == WHITE ? BLACK_KING : WHITE_KING;
47 constexpr PieceColor opponentColor = !color;
48 bool onlyKingMoves = false;
49 const uint64_t ownPieces = board.getColorBitboard<color>();
50 const uint64_t opponentKingBB = board.getBitboard<opponentKing>();
51 uint64_t genMask = ~(ownPieces | opponentKingBB);
52
53 if (type == QSEARCH) {
54 const uint64_t opponentPieces = board.getColorBitboard<opponentColor>();
55
56 genMask &= opponentPieces;
57 } else if (type == EVASIONS) {
58 const Square kingSquare = bitboardToSquare(board.getBitboard<ownKing>());
59 const uint64_t attackers = board.getSquareAttackersByColor<opponentColor>(kingSquare);
60
61 if (popcnt(attackers) == 1) {
62 const Square attackerSquare = bitboardToSquare(attackers);
63 const PieceType attackerType = getPieceType(board.getPieceOnSquare(attackerSquare));
64 uint64_t evasionsMask = attackers;
65
66 if (isSlidingPiece(attackerType)) {
67 const uint64_t squaresBetween = getSquaresBetween(attackerSquare, kingSquare);
68
69 evasionsMask |= squaresBetween;
70 }
71
72 genMask = evasionsMask;
73 } else {
74 // If the king is in double check, only king moves are legal
75 onlyKingMoves = true;
76 }
77 }
78
79 if (!onlyKingMoves) {
80 generatePawnMoves<color, type>(board, moves, genMask);
81 generateKnightMoves<color, type>(board, moves, genMask);
82 generateBishopMoves<color, type>(board, moves, genMask);
83 generateRookMoves<color, type>(board, moves, genMask);
84 generateQueenMoves<color, type>(board, moves, genMask);
85 }
86
87 if (type == EVASIONS) {
88 // We generate all king moves when in check, so we don't use genMask
89 generateKingMoves<color, type>(board, moves, ~(ownPieces | opponentKingBB));
90 } else {
91 generateKingMoves<color, type>(board, moves, genMask);
92 }
93}

103template <PieceColor color, GenerationType type>
104void generatePawnMoves(const Board& board, MoveList& moves, const uint64_t genMask) {
105 constexpr Piece pawn = color == WHITE ? WHITE_PAWN : BLACK_PAWN;
106 constexpr PieceColor opponentColor = !color;
107
108 const uint64_t pawnBB = board.getBitboard<pawn>();
109 const uint64_t emptyBB = board.getEmptyBitboard();
110 const uint64_t opponentPieces = board.getColorBitboard<opponentColor>();
111 uint64_t pawnSinglePushes;
112 uint64_t pawnDoublePushes;
113 uint64_t pawnWestAttacks;
114 uint64_t pawnEastAttacks;
115
116 if constexpr (color == WHITE) {
117 pawnSinglePushes = whitePawnSinglePush(pawnBB, emptyBB);
118 pawnDoublePushes = whitePawnDoublePush(pawnBB, emptyBB);
119 pawnWestAttacks = whitePawnWestAttacks(pawnBB);
120 pawnEastAttacks = whitePawnEastAttacks(pawnBB);
121 } else {
122 pawnSinglePushes = blackPawnSinglePush(pawnBB, emptyBB);
123 pawnDoublePushes = blackPawnDoublePush(pawnBB, emptyBB);
124 pawnWestAttacks = blackPawnWestAttacks(pawnBB);
125 pawnEastAttacks = blackPawnEastAttacks(pawnBB);
126 }
127
128 uint64_t enPassantMask = squareToBitboard(board.getEnPassantSquare());
129
130 if constexpr (color == WHITE) {
131 enPassantMask &= RANK_6;
132 } else {
133 enPassantMask &= RANK_3;
134 }
135
136 pawnSinglePushes &= genMask;
137 pawnDoublePushes &= genMask;
138 pawnWestAttacks &= (opponentPieces | enPassantMask) & genMask;
139 pawnEastAttacks &= (opponentPieces | enPassantMask) & genMask;
140
141 constexpr Direction fromPushDirection = color == WHITE ? NORTH : SOUTH;
142 constexpr Direction fromSqWestAttackDirection = color == WHITE ? NORTH_WEST : SOUTH_WEST;
143 constexpr Direction fromSqEastAttackDirection = color == WHITE ? NORTH_EAST : SOUTH_EAST;
144 constexpr uint64_t promotionRank = color == WHITE ? RANK_8 : RANK_1;
145
146 while (pawnSinglePushes) {
147 const uint8_t squareTo = popLsb(pawnSinglePushes);
148 const uint64_t squareToBB = squareToBitboard(squareTo);
149 const uint8_t squareFrom = squareTo - fromPushDirection;
150
151 if (squareToBB & promotionRank) {
152 for (const PromotionPiece promotionPiece : {QUEEN_PROMOTION, ROOK_PROMOTION, BISHOP_PROMOTION,
153 KNIGHT_PROMOTION}) {
154 const Move move = encodeMove(squareFrom, squareTo, promotionPiece);
155 moves.moves[moves.size] = move;
156 moves.size++;
157 }
158 } else {
159 const Move move = encodeMove(squareFrom, squareTo);
160
161 moves.moves[moves.size] = move;
162 moves.size++;
163 }
164 }
165
166 while (pawnDoublePushes) {
167 const uint8_t squareTo = popLsb(pawnDoublePushes);
168 const uint8_t squareFrom = squareTo - fromPushDirection - fromPushDirection;
169 const Move move = encodeMove(squareFrom, squareTo);
170
171 moves.moves[moves.size] = move;
172 moves.size++;
173 }
174
175 while (pawnWestAttacks) {
176 const uint8_t squareTo = popLsb(pawnWestAttacks);
177 const uint64_t squareToBB = squareToBitboard(squareTo);
178 const uint8_t squareFrom = squareTo - fromSqWestAttackDirection;
179
180 if (squareTo == board.getEnPassantSquare() && squareToBB & enPassantMask) {
181 const Move move = encodeMove(squareFrom, squareTo, EN_PASSANT);
182 moves.moves[moves.size] = move;
183 moves.size++;
184 } else {
185 if (squareToBB & promotionRank) {
186 for (const PromotionPiece promotionPiece : {QUEEN_PROMOTION, ROOK_PROMOTION, BISHOP_PROMOTION,
187 KNIGHT_PROMOTION}) {
188 const Move move = encodeMove(squareFrom, squareTo, promotionPiece);
189 moves.moves[moves.size] = move;
190 moves.size++;
191 }
192 } else {
193 const Move move = encodeMove(squareFrom, squareTo);
194 moves.moves[moves.size] = move;
195 moves.size++;
196 }
197 }
198 }
199
200 while (pawnEastAttacks) {
201 const uint8_t squareTo = popLsb(pawnEastAttacks);
202 const uint64_t squareToBB = squareToBitboard(squareTo);
203 const uint8_t squareFrom = squareTo - fromSqEastAttackDirection;
204
205 if (squareTo == board.getEnPassantSquare() && squareToBB & enPassantMask) {
206 const Move move = encodeMove(squareFrom, squareTo, EN_PASSANT);
207 moves.moves[moves.size] = move;
208 moves.size++;
209 } else {
210 if (squareToBB & promotionRank) {
211 for (const PromotionPiece promotionPiece : {QUEEN_PROMOTION, ROOK_PROMOTION, BISHOP_PROMOTION,
212 KNIGHT_PROMOTION}) {
213 const Move move = encodeMove(squareFrom, squareTo, promotionPiece);
214 moves.moves[moves.size] = move;
215 moves.size++;
216 }
217 } else {
218 const Move move = encodeMove(squareFrom, squareTo);
219 moves.moves[moves.size] = move;
220 moves.size++;
221 }
222 }
223 }
224}

234template <PieceColor color, GenerationType type>
235void generateKnightMoves(const Board& board, MoveList& moves, const uint64_t genMask) {
236 constexpr Piece knight = color == WHITE ? WHITE_KNIGHT : BLACK_KNIGHT;
237 uint64_t knightBB = board.getBitboard<knight>();
238
239 while (knightBB) {
240 const uint8_t fromSquare = popLsb(knightBB);
241 const uint64_t attacks = getKnightAttacks(fromSquare);
242 uint64_t genBB = attacks & genMask;
243
244 while (genBB) {
245 const uint8_t toSquare = popLsb(genBB);
246 const Move move = encodeMove(fromSquare, toSquare);
247
248 moves.moves[moves.size] = move;
249 moves.size++;
250 }
251 }

346template <PieceColor color, GenerationType type>
347void generateKingMoves(const Board& board, MoveList& moves, const uint64_t genMask) {
348 constexpr Piece king = color == WHITE ? WHITE_KING : BLACK_KING;
349 uint64_t kingBB = board.getBitboard<king>();
350 const uint8_t fromSquare = popLsb(kingBB);
351 uint64_t genBB = getKingAttacks(fromSquare) & genMask;
352
353 while (genBB) {
354 const uint8_t toSquare = popLsb(genBB);
355 const Move move = encodeMove(fromSquare, toSquare);
356
357 moves.moves[moves.size] = move;
358 moves.size++;
359 }
360
361 const uint8_t castlingRights = board.getCastlingRights();
362
363 if constexpr (color == WHITE) {
364 if (castlingRights & WHITE_KINGSIDE && board.canCastle<WHITE_KINGSIDE>()) {
365 const Move move = encodeMove(E1, G1, CASTLING);
366
367 moves.moves[moves.size] = move;
368 moves.size++;
369 }
370
371 if (castlingRights & WHITE_QUEENSIDE && board.canCastle<WHITE_QUEENSIDE>()) {
372 const Move move = encodeMove(E1, C1, CASTLING);
373
374 moves.moves[moves.size] = move;
375 moves.size++;
376 }
377 } else if constexpr (color == BLACK) {
378 if (castlingRights & BLACK_KINGSIDE && board.canCastle<BLACK_KINGSIDE>()) {
379 const Move move = encodeMove(E8, G8, CASTLING);
380
381 moves.moves[moves.size] = move;
382 moves.size++;
383 }
384
385 if (castlingRights & BLACK_QUEENSIDE && board.canCastle<BLACK_QUEENSIDE>()) {
386 const Move move = encodeMove(E8, C8, CASTLING);
387
388 moves.moves[moves.size] = move;
389 moves.size++;
390 }
391 }
