56template <PieceColor movedColor>
57bool Board::isPositionLegal() const {
58 constexpr PieceColor opponentColor = !movedColor;
59 constexpr Piece king = movedColor == WHITE ? WHITE_KING : BLACK_KING;
60 const uint64_t kingBB = getBitboard<king>();
61 const Square kingSquare = bitboardToSquare(kingBB);
62 const Move lastMove = getLastMove();
63 const MoveType lastMoveType = getMoveType(lastMove);
64
65 if (lastMoveType == CASTLING) {
66 const uint8_t fromSquare = getFromSquare(lastMove);
67 const uint8_t toSquare = getToSquare(lastMove);
68 const uint64_t fromSquareAttacks = getSquareAttackersByColor<opponentColor>(fromSquare);
69
70 if (fromSquareAttacks) {
71 // King was in check before castling
72 return false;
73 }
74
75 uint64_t castlingPath = 0;
76
77 if (toSquare == G1) {
78 castlingPath = WHITE_KINGSIDE_CASTLE_PATH;
79 } else if (toSquare == C1) {
80 castlingPath = WHITE_QUEENSIDE_CASTLE_PATH;
81 } else if (toSquare == G8) {
82 castlingPath = BLACK_KINGSIDE_CASTLE_PATH;
83 } else if (toSquare == C8) {
84 castlingPath = BLACK_QUEENSIDE_CASTLE_PATH;
85 }
86
87 while (castlingPath) {
88 const uint8_t square = popLsb(castlingPath);
89 const uint64_t attackers = getSquareAttackersByColor<opponentColor>(square);
90
91 if (attackers) {
92 return false;
93 }
94 }
95 }
96
97 return !getSquareAttackersByColor<opponentColor>(kingSquare);
98}

108template <CastlingRights side>
109bool Board::canCastle() const {
110 assert(side != WHITE_CASTLING && side != BLACK_CASTLING);
111
112 if (!(castlingRights & side)) {
113 return false;
114 }
115
116 uint64_t castlingPath = 0;
117
118 if constexpr (side == WHITE_KINGSIDE) {
119 castlingPath = WHITE_KINGSIDE_CASTLE_UNOCCUPIED;
120 } else if constexpr (side == WHITE_QUEENSIDE) {
121 castlingPath = WHITE_QUEENSIDE_CASTLE_UNOCCUPIED;
122 } else if constexpr (side == BLACK_KINGSIDE) {
123 castlingPath = BLACK_KINGSIDE_CASTLE_UNOCCUPIED;
124 } else if constexpr (side == BLACK_QUEENSIDE) {
125 castlingPath = BLACK_QUEENSIDE_CASTLE_UNOCCUPIED;
126 }
127
128 if (occupied & castlingPath) {
129 return false;
130 }
131
132 return true;
133}

145uint64_t Board::getSquareAttackers(const uint8_t square) const {
146 assert(square < SQUARES);
147 const uint64_t knights = getBitboard<WHITE_KNIGHT>() | getBitboard<BLACK_KNIGHT>();
148 const uint64_t kings = getBitboard<WHITE_KING>() | getBitboard<BLACK_KING>();
149 uint64_t bishopsQueens = getBitboard<WHITE_QUEEN>() | getBitboard<BLACK_QUEEN>();
150 uint64_t rooksQueens = getBitboard<WHITE_QUEEN>() | getBitboard<BLACK_QUEEN>();
151 rooksQueens |= getBitboard<WHITE_ROOK>() | getBitboard<BLACK_ROOK>();
152 bishopsQueens |= getBitboard<WHITE_BISHOP>() | getBitboard<BLACK_BISHOP>();
153
154 return (getPawnAttacks<WHITE>(square) & getBitboard<BLACK_PAWN>())
155 | (getPawnAttacks<BLACK>(square) & getBitboard<WHITE_PAWN>())
156 | (getKnightAttacks(square) & knights)
157 | (getKingAttacks(square) & kings)
158 | (getBishopAttacks(square, occupied) & bishopsQueens)
159 | (getRookAttacks(square, occupied) & rooksQueens);
160}

211void Board::makeMove(const Move& move) {
212 const uint8_t fromSquare = getFromSquare(move);
213 const uint8_t toSquare = getToSquare(move);
214 const MoveType moveType = getMoveType(move);
215 Piece movedPiece = getPieceOnSquare(fromSquare);
216 const PieceType movedPieceType = getPieceType(movedPiece);
217 const Piece capturedPiece = getPieceOnSquare(toSquare);
218
219 history[ply].move = move;
220 history[ply].capturedPiece = capturedPiece;
221 history[ply].enPassantSquare = enPassantSquare;
222 zobristHash ^= getZobristConstant(ZOBRIST_EN_PASSANT_START_INDEX + (enPassantSquare % 8));
223 enPassantSquare = 0;
224 history[ply].castlingRights = castlingRights;
225 history[ply].zobristHash = zobristHash;
226 history[ply].halfMoveClock = halfMoveClock;
227
228 halfMoveClock += 1;
229
230 if (capturedPiece != EMPTY) {
231 removePiece(capturedPiece, toSquare);
232 halfMoveClock = 0;
233
234 if (capturedPiece == WHITE_ROOK) {
235 if (toSquare == A1) {
236 if (castlingRights & WHITE_QUEENSIDE) {
237 castlingRights &= ~WHITE_QUEENSIDE;
238 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_WHITE_QUEENSIDE_INDEX);
239 }
240 } else if (toSquare == H1) {
241 if (castlingRights & WHITE_KINGSIDE) {
242 castlingRights &= ~WHITE_KINGSIDE;
243 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_WHITE_KINGSIDE_INDEX);
244 }
245 }
246 } else if (capturedPiece == BLACK_ROOK) {
247 if (toSquare == A8) {
248 if (castlingRights & BLACK_QUEENSIDE) {
249 castlingRights &= ~BLACK_QUEENSIDE;
250 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_BLACK_QUEENSIDE_INDEX);
251 }
252 } else if (toSquare == H8) {
253 if (castlingRights & BLACK_KINGSIDE) {
254 castlingRights &= ~BLACK_KINGSIDE;
255 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_BLACK_KINGSIDE_INDEX);
256 }
257 }
258 }
259 }
260
261 removePiece(movedPiece, fromSquare);
262
263 if (moveType == PROMOTION) {
264 const PieceColor color = getPieceColor(movedPiece);
265 const PromotionPiece promotionPieceType = getPromotionPiece(move);
266 const Piece promotionPiece = getPieceFromPromotionPiece(promotionPieceType, color);
267
268 setPiece(promotionPiece, toSquare);
269 } else {
270 setPiece(movedPiece, toSquare);
271 }
272
273 if (moveType == EN_PASSANT) {
274 if (sideToMove == WHITE) {
275 removePiece(BLACK_PAWN, toSquare + SOUTH);
276 } else {
277 removePiece(WHITE_PAWN, toSquare + NORTH);
278 }
279 } else if (moveType == CASTLING) {
280 if (toSquare == G1) {
281 removePiece(WHITE_ROOK, H1);
282 setPiece(WHITE_ROOK, F1);
283 } else if (toSquare == C1) {
284 removePiece(WHITE_ROOK, A1);
285 setPiece(WHITE_ROOK, D1);
286 } else if (toSquare == G8) {
287 removePiece(BLACK_ROOK, H8);
288 setPiece(BLACK_ROOK, F8);
289 } else if (toSquare == C8) {
290 removePiece(BLACK_ROOK, A8);
291 setPiece(BLACK_ROOK, D8);
292 }
293
294 if (sideToMove == WHITE) {
295 if (castlingRights & WHITE_KINGSIDE) {
296 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_WHITE_KINGSIDE_INDEX);
297 }
298
299 if (castlingRights & WHITE_QUEENSIDE) {
300 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_WHITE_QUEENSIDE_INDEX);
301 }
302
303 castlingRights &= ~WHITE_CASTLING;
304 } else {
305 if (castlingRights & BLACK_KINGSIDE) {
306 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_BLACK_KINGSIDE_INDEX);
307 }
308
309 if (castlingRights & BLACK_QUEENSIDE) {
310 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_BLACK_QUEENSIDE_INDEX);
311 }
312
313 castlingRights &= ~BLACK_CASTLING;
314 }
315 }
316
317 if (movedPiece == WHITE_KING) {
318 if (castlingRights & WHITE_KINGSIDE) {
319 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_WHITE_KINGSIDE_INDEX);
320 }
321
322 if (castlingRights & WHITE_QUEENSIDE) {
323 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_WHITE_QUEENSIDE_INDEX);
324 }
325
326 castlingRights &= ~WHITE_CASTLING;
327 } else if (movedPiece == BLACK_KING) {
328 if (castlingRights & BLACK_KINGSIDE) {
329 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_BLACK_KINGSIDE_INDEX);
330 }
331
332 if (castlingRights & BLACK_QUEENSIDE) {
333 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_BLACK_QUEENSIDE_INDEX);
334 }
335
336 castlingRights &= ~BLACK_CASTLING;
337 } else if (movedPiece == WHITE_ROOK) {
338 if (fromSquare == A1) {
339 if (castlingRights & WHITE_QUEENSIDE) {
340 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_WHITE_QUEENSIDE_INDEX);
341 castlingRights &= ~WHITE_QUEENSIDE;
342 }
343 } else if (fromSquare == H1) {
344 if (castlingRights & WHITE_KINGSIDE) {
345 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_WHITE_KINGSIDE_INDEX);
346 castlingRights &= ~WHITE_KINGSIDE;

358 castlingRights &= ~BLACK_KINGSIDE;
359 }
360 }
361 }
362
363 if (movedPieceType == PAWN) {
364 halfMoveClock = 0;
365
366 if ((fromSquare ^ toSquare) == 16) {
367 if (sideToMove == WHITE) {
368 enPassantSquare = toSquare + SOUTH;
369 zobristHash ^= getZobristConstant(ZOBRIST_EN_PASSANT_START_INDEX + (enPassantSquare % 8));
370 } else {
371 enPassantSquare = toSquare + NORTH;
372 zobristHash ^= getZobristConstant(ZOBRIST_EN_PASSANT_START_INDEX + (enPassantSquare % 8));
373 }
374 }
375 }
376
377 if (sideToMove == BLACK) {
378 fullmoveClock += 1;
379 }
380
381 sideToMove = !sideToMove;
382 zobristHash ^= getZobristConstant(ZOBRIST_SIDE_TO_MOVE_INDEX);
383
384 assert(ply >= 0 && ply < MAX_PLY);
385 ply++;
386 assert(ply >= 0 && ply < MAX_PLY);
387}

392void Board::unmakeMove() {
393 ply--;
394 assert(ply >= 0 && ply < MAX_PLY);
395 const BoardState& state = history[ply];
396 const uint8_t fromSquare = getFromSquare(state.move);
397 const uint8_t toSquare = getToSquare(state.move);
398 const MoveType moveType = getMoveType(state.move);
399 Piece movedPiece = getPieceOnSquare(toSquare);
400 PieceColor movedColor = getPieceColor(movedPiece);
401
402 removePiece(movedPiece, toSquare);
403
404 if (moveType == PROMOTION) {
405 const PieceColor color = getPieceColor(movedPiece);
406 movedPiece = static_cast<Piece>(WHITE_PAWN + color);
407 }
408
409 setPiece(movedPiece, fromSquare);
410
411 if (state.capturedPiece != EMPTY) {
412 setPiece(state.capturedPiece, toSquare);
413 }
414
415 if (moveType == EN_PASSANT) {
416 const PieceColor movedPieceColor = getPieceColor(movedPiece);
417
418 if (movedPieceColor == WHITE) {
419 setPiece(BLACK_PAWN, toSquare + SOUTH);
420 } else {
421 setPiece(WHITE_PAWN, toSquare + NORTH);
422 }
423 }
424
425 if (moveType == CASTLING) {
426 if (toSquare == G1) {
427 removePiece(WHITE_ROOK, F1);
428 setPiece(WHITE_ROOK, H1);
429 } else if (toSquare == C1) {
430 removePiece(WHITE_ROOK, D1);
431 setPiece(WHITE_ROOK, A1);
432 } else if (toSquare == G8) {
433 removePiece(BLACK_ROOK, F8);
434 setPiece(BLACK_ROOK, H8);
435 } else if (toSquare == C8) {
436 removePiece(BLACK_ROOK, D8);
437 setPiece(BLACK_ROOK, A8);
438 }
439 }
440
441 if (movedColor == BLACK) {
442 fullmoveClock -= 1;
443 }
444
445 this->halfMoveClock = state.halfMoveClock;
446 this->sideToMove = !sideToMove;
447 this->enPassantSquare = state.enPassantSquare;
448 this->castlingRights = state.castlingRights;
449 this->zobristHash = state.zobristHash;
450}

457void Board::setPieceFromFENChar(const char character, const uint8_t square) {
458 assert(square < SQUARES);
459 // Uppercase char = white, lowercase = black
460 switch (character) {
461 case 'P':
462 setPiece(WHITE_PAWN, square);
463 break;
464 case 'p':
465 setPiece(BLACK_PAWN, square);
466 break;
467 case 'N':
468 setPiece(WHITE_KNIGHT, square);
469 break;
470 case 'n':
471 setPiece(BLACK_KNIGHT, square);
472 break;
473 case 'B':
474 setPiece(WHITE_BISHOP, square);
475 break;
476 case 'b':
477 setPiece(BLACK_BISHOP, square);
478 break;
479 case 'R':
480 setPiece(WHITE_ROOK, square);
481 break;
482 case 'r':
483 setPiece(BLACK_ROOK, square);
484 break;
485 case 'Q':
486 setPiece(WHITE_QUEEN, square);
487 break;
488 case 'q':
489 setPiece(BLACK_QUEEN, square);
490 break;
491 case 'K':
492 setPiece(WHITE_KING, square);
493 break;
494 case 'k':
495 setPiece(BLACK_KING, square);
496 break;
497 default:
498 break;
499 }

507bool Board::setFromFEN(const std::string_view fen) {
508 uint8_t index = A8;
509 int spaces = 0;
510
511 reset();
512
513 for (const char character : fen) {
514 if (character == ' ') {
515 spaces++;
516 continue;
517 }
518
519 if (character == ',') {
520 break;
521 }
522
523 if (spaces == 0) {
524 if (character == '/') {
525 index -= 16;
526 continue;
527 }
528
529 if (character >= '1' && character <= '8') {
530 index += character - '0';
531 continue;
532 }
533
534 if (character >= 'A' && character <= 'z') {
535 setPieceFromFENChar(character, index);
536 index++;
537 } else {
538 return false;
539 }
540 }
541
542 if (spaces == 1) {
543 if (tolower(character) == 'w') {
544 sideToMove = WHITE;
545 } else if (tolower(character) == 'b') {
546 sideToMove = BLACK;
547 zobristHash ^= getZobristConstant(ZOBRIST_SIDE_TO_MOVE_INDEX);
548 } else {
549 return false;
550 }
551 }
552
553 if (spaces == 2) {
554 if (character == '-') {
555 continue;
556 } else if (character == 'K') {
557 castlingRights |= WHITE_KINGSIDE;
558 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_WHITE_KINGSIDE_INDEX);
559 continue;
560 }
561
562 if (character == 'Q') {
563 castlingRights |= WHITE_QUEENSIDE;
564 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_WHITE_QUEENSIDE_INDEX);
565 continue;
566 }
567
568 if (character == 'k') {
569 castlingRights |= BLACK_KINGSIDE;
570 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_BLACK_KINGSIDE_INDEX);
571 continue;
572 }
573
574 if (character == 'q') {
575 castlingRights |= BLACK_QUEENSIDE;
576 zobristHash ^= getZobristConstant(ZOBRIST_CASTLING_BLACK_QUEENSIDE_INDEX);
577 continue;
578 }
579
580 return false;
581 }
582
583 if (spaces == 3) {
584 if (character == '-') {
585 continue;
586 }
587
588 if (tolower(character) < 'a' || tolower(character) > 'h') {
589 continue;
590 }
591
592 const int8_t file = tolower(character) - 'a';
593 const int8_t rank = (!sideToMove) == WHITE ? 2 : 5;
594
595 if (file < 0 || file > 7) {
596 return false;
597 }
598
599 enPassantSquare = rank * 8 + file;
600 zobristHash ^= getZobristConstant(ZOBRIST_EN_PASSANT_START_INDEX + (enPassantSquare % 8));
601 index += 2;
602 }
603
604 if (spaces == 4) {
605 halfMoveClock = character - '0';
606 }
607
608 if (spaces == 5) {
609 fullmoveClock = character - '0';
610 }
611 }
612
613 return true;
614}
