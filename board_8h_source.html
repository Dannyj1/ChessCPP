235 template <Piece piece>
236 void setPiece(const uint8_t square) {
237 assert(piece != Piece::EMPTY);
238 assert(board[square] == Piece::EMPTY);
239 const uint64_t squareBB = squareToBitboard(square);
240
241 board[square] = piece;
242 bitboards[piece] |= squareBB;
243 occupied |= squareBB;
244 colorBoards[getPieceColor(piece)] |= squareBB;
245
246 const int zobristIndex = ZOBRIST_PIECE_START_INDEX + static_cast<int>(piece) * SQUARES + square;
247 zobristHash ^= getZobristConstant(zobristIndex);
248 }

354 template <PieceColor color>

356 constexpr PieceColor opponentColor = !color;
357 const uint64_t kingBB = getPieceBoard<color == WHITE ? WHITE_KING : BLACK_KING>();
