26void TranspositionTable::savePosition(const uint64_t zobristHash, const int8_t depth, const int ply, int score,
27 const Move bestMove, const TTNodeType nodeType) const {
28 const uint64_t index = zobristHash & hashSize;
29 TTEntry* entry = &transpositionTable[index];
30
31 // Only replace the entry if:
32 // 1. Validation hash is 0 (the entry is empty)
33 // 1. Depth > 0 (the new node is a pvSearch node, can replace any node)
34 // 2. The entries' depth < 0 (the entry is a qSearch node, can be replaced by any node)
35 if (entry->validationHash == 0 || depth > 0 || entry->depth < 0) {
36 if (score >= (MATE_SCORE - MAX_PLY)) {
37 score += ply;
38 } else if (score <= (-MATE_SCORE + MAX_PLY)) {
39 score -= ply;
40 }
41
42 score = std::clamp(score, INT16_MIN, INT16_MAX);
43
44 entry->validationHash = zobristHash >> 32;
45 entry->depth = depth;
46 entry->bestMove = bestMove;
47 entry->score = score;
48 entry->nodeType = nodeType;
49 }
50}

52int16_t TranspositionTable::probePosition(const uint64_t zobristHash, const int8_t depth, const int alpha,
53 const int beta, const int ply) const {
54 const uint64_t index = zobristHash & hashSize;
55 const uint32_t validationHash = zobristHash >> 32;
56 const TTEntry* entry = &transpositionTable[index];
57
58 if (entry->validationHash == validationHash && entry->depth >= depth) {
59 bool returnScore = false;
60
61 if (entry->nodeType == EXACT) {
62 returnScore = true;
63 } else if (entry->nodeType == ALPHA) {
64 if (entry->score <= alpha) {
65 returnScore = true;
66 }
67 } else if (entry->nodeType == BETA) {
68 if (entry->score >= beta) {
69 returnScore = true;
70 }
71 }
72
73 if (returnScore) {
74 int adjustedScore = entry->score;
75
76 if (adjustedScore >= MATE_SCORE) {
77 adjustedScore -= ply;
78 } else if (adjustedScore <= -MATE_SCORE) {
79 adjustedScore += ply;
80 }
81
82 return adjustedScore;
83 }
84 }
85
86 return NO_TT_SCORE;
87}

89TTEntry* TranspositionTable::getEntry(const uint64_t zobristHash) const {
90 const uint64_t index = zobristHash & hashSize;
91 TTEntry* entry = &transpositionTable[index];
92
93 // Check validation hash to avoid hash collisions
94 if (entry->validationHash == (zobristHash >> 32)) {
95 return entry;
96 }
97
98 return nullptr;
99}

101void TranspositionTable::setTableSize(int megaBytes) {
102 if ((megaBytes & (megaBytes - 1)) != 0) {
103 megaBytes = 1 << static_cast<int>(log2(megaBytes));
104 }
105
106 const uint64_t byteSize = megaBytes * 1024 * 1024;
107 const uint64_t entryCount = byteSize / sizeof(TTEntry);
108
109 delete[] transpositionTable;
110 transpositionTable = new TTEntry[entryCount]{};
111 hashSize = entryCount - 1;
112
113 std::fill_n(transpositionTable, entryCount, TTEntry{});
114}

121template <PieceColor color>
122void TranspositionTable::updateHistory(const Move move, const int value) {
123 const Square from = getFromSquare(move);
124 const Square to = getToSquare(move);
125 const int clampedValue = std::clamp(value, -MAX_HISTORY, MAX_HISTORY);
126
127 history[color][from][to] += clampedValue - history[color][from][to] * std::abs(clampedValue) / MAX_HISTORY;
128}
