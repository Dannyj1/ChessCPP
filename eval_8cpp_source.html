38int Evaluation::evaluate() {
39 const int phase = calculatePhase();
40 const int modifier = board.getSideToMove() == WHITE ? 1 : -1;
41
42 initializeEvalData();
43
44 evaluatePieces();
45 evaluateMobility();
46
47 const int whiteScore = ((whiteMidgameScore * (256 - phase)) + (whiteEndgameScore * phase)) / 256;
48 const int blackScore = ((blackMidgameScore * (256 - phase)) + (blackEndgameScore * phase)) / 256;
49
50 return (whiteScore - blackScore) * modifier;
51}
52
53constexpr int knightPhase = 1;
54constexpr int bishopPhase = 1;
55constexpr int rookPhase = 2;
56constexpr int queenPhase = 4;
57constexpr int totalPhase = knightPhase * 4 + bishopPhase * 4 + rookPhase * 4 + queenPhase * 2;
58
68int Evaluation::calculatePhase() const {
69 int phase = totalPhase;
70
71 phase -= popcnt(board.getPieceBoard<WHITE_KNIGHT>() | board.getPieceBoard<BLACK_KNIGHT>()) * knightPhase;
72 phase -= popcnt(board.getPieceBoard<WHITE_BISHOP>() | board.getPieceBoard<BLACK_BISHOP>()) * bishopPhase;
73 phase -= popcnt(board.getPieceBoard<WHITE_ROOK>() | board.getPieceBoard<BLACK_ROOK>()) * rookPhase;
74 phase -= popcnt(board.getPieceBoard<WHITE_QUEEN>() | board.getPieceBoard<BLACK_QUEEN>()) * queenPhase;
75
76 return (phase * 256 + (totalPhase / 2)) / totalPhase;
77}

79void Evaluation::evaluateMobility() {
80 // Mobility
81 uint64_t whiteKnightMobility = evalData.attacksByPiece[WHITE_KNIGHT];
82 uint64_t blackKnightMobility = evalData.attacksByPiece[BLACK_KNIGHT];
83 uint64_t whiteBishopMobility = evalData.attacksByPiece[WHITE_BISHOP];
84 uint64_t blackBishopMobility = evalData.attacksByPiece[BLACK_BISHOP];
85 uint64_t whiteRookMobility = evalData.attacksByPiece[WHITE_ROOK];
86 uint64_t blackRookMobility = evalData.attacksByPiece[BLACK_ROOK];
87 uint64_t whiteQueenMobility = evalData.attacksByPiece[WHITE_QUEEN];
88 uint64_t blackQueenMobility = evalData.attacksByPiece[BLACK_QUEEN];
89
90 int whiteKnightMobilityCount = popcnt(whiteKnightMobility);
91 int blackKnightMobilityCount = popcnt(blackKnightMobility);
92 int whiteBishopMobilityCount = popcnt(whiteBishopMobility);
93 int blackBishopMobilityCount = popcnt(blackBishopMobility);
94 int whiteRookMobilityCount = popcnt(whiteRookMobility);
95 int blackRookMobilityCount = popcnt(blackRookMobility);
96 int whiteQueenMobilityCount = popcnt(whiteQueenMobility);
97 int blackQueenMobilityCount = popcnt(blackQueenMobility);
98
99 whiteMidgameScore += whiteKnightMobilityCount * getEvalFeatureValue(MOBILITY_MIDGAME_KNIGHT_VALUE);
100 blackMidgameScore += blackKnightMobilityCount * getEvalFeatureValue(MOBILITY_MIDGAME_KNIGHT_VALUE);
101 whiteMidgameScore += whiteBishopMobilityCount * getEvalFeatureValue(MOBILITY_MIDGAME_BISHOP_VALUE);
102 blackMidgameScore += blackBishopMobilityCount * getEvalFeatureValue(MOBILITY_MIDGAME_BISHOP_VALUE);
103 whiteMidgameScore += whiteRookMobilityCount * getEvalFeatureValue(MOBILITY_MIDGAME_ROOK_VALUE);
104 blackMidgameScore += blackRookMobilityCount * getEvalFeatureValue(MOBILITY_MIDGAME_ROOK_VALUE);
105 whiteMidgameScore += whiteQueenMobilityCount * getEvalFeatureValue(MOBILITY_MIDGAME_QUEEN_VALUE);
106 blackMidgameScore += blackQueenMobilityCount * getEvalFeatureValue(MOBILITY_MIDGAME_QUEEN_VALUE);
107
108 whiteEndgameScore += whiteKnightMobilityCount * getEvalFeatureValue(MOBILITY_ENDGAME_KNIGHT_VALUE);
109 blackEndgameScore += blackKnightMobilityCount * getEvalFeatureValue(MOBILITY_ENDGAME_KNIGHT_VALUE);
110 whiteEndgameScore += whiteBishopMobilityCount * getEvalFeatureValue(MOBILITY_ENDGAME_BISHOP_VALUE);
111 blackEndgameScore += blackBishopMobilityCount * getEvalFeatureValue(MOBILITY_ENDGAME_BISHOP_VALUE);
112 whiteEndgameScore += whiteRookMobilityCount * getEvalFeatureValue(MOBILITY_ENDGAME_ROOK_VALUE);
113 blackEndgameScore += blackRookMobilityCount * getEvalFeatureValue(MOBILITY_ENDGAME_ROOK_VALUE);
114 whiteEndgameScore += whiteQueenMobilityCount * getEvalFeatureValue(MOBILITY_ENDGAME_QUEEN_VALUE);
115 blackEndgameScore += blackQueenMobilityCount * getEvalFeatureValue(MOBILITY_ENDGAME_QUEEN_VALUE);
116}

118void Evaluation::evaluatePieces() {
119 uint64_t pieces = board.getOccupiedBitboard();
120
121 while (pieces) {
122 const Square square = static_cast<Square>(popLsb(pieces));
123 const Piece piece = board.getPieceOnSquare(square);
124 const PieceColor color = getPieceColor(piece);
125
126 const int midgamePst = getMidgamePstEvalFeature(piece, square);
127 const int endgamePst = getEndgamePstEvalFeature(piece, square);
128
129 if (color == WHITE) {
130 whiteMidgameScore += midgamePst;
131 whiteEndgameScore += endgamePst;
132 } else {
133 blackMidgameScore += midgamePst;
134 blackEndgameScore += endgamePst;
135 }
136 }
137}

142void Evaluation::initializeEvalData() {
143 const uint64_t occupiedBB = board.getOccupiedBitboard();
144 uint64_t pieces = occupiedBB;
145 const uint64_t whitePieces = board.getColorBitboard<WHITE>();
146 const uint64_t blackPieces = board.getColorBitboard<BLACK>();
147
148 while (pieces) {
149 const uint64_t pieceBB = popLsb(pieces);
150 const Square square = bitboardToSquare(pieceBB);
151 const Piece piece = board.getPieceOnSquare(square);
152
153 switch (piece) {
154 case WHITE_PAWN:
155 evalData.attacksFrom[square] = getPawnAttacks<WHITE>(square) & ~whitePieces;
156 evalData.attacksByColor[WHITE] |= evalData.attacksFrom[square];
157 evalData.attacksByPiece[WHITE_PAWN] |= evalData.attacksFrom[square];
158 break;
159 case BLACK_PAWN:
160 evalData.attacksFrom[square] = getPawnAttacks<BLACK>(square) & ~blackPieces;
161 evalData.attacksByColor[BLACK] |= evalData.attacksFrom[square];
162 evalData.attacksByPiece[BLACK_PAWN] |= evalData.attacksFrom[square];
163 break;
164 case WHITE_KNIGHT:
165 evalData.attacksFrom[square] = getKnightAttacks(square) & ~whitePieces;
166 evalData.attacksByColor[WHITE] |= evalData.attacksFrom[square];
167 evalData.attacksByPiece[WHITE_KNIGHT] |= evalData.attacksFrom[square];
168 break;
169 case BLACK_KNIGHT:
170 evalData.attacksFrom[square] = getKnightAttacks(square) & ~blackPieces;
171 evalData.attacksByColor[BLACK] |= evalData.attacksFrom[square];
172 evalData.attacksByPiece[BLACK_KNIGHT] |= evalData.attacksFrom[square];
173 break;
174 case WHITE_BISHOP:
175 evalData.attacksFrom[square] = getBishopAttacks(square, occupiedBB) & ~whitePieces;
176 evalData.attacksByColor[WHITE] |= evalData.attacksFrom[square];
177 evalData.attacksByPiece[WHITE_BISHOP] |= evalData.attacksFrom[square];
178 break;
179 case BLACK_BISHOP:
180 evalData.attacksFrom[square] = getBishopAttacks(square, occupiedBB) & ~blackPieces;
181 evalData.attacksByColor[BLACK] |= evalData.attacksFrom[square];
182 evalData.attacksByPiece[BLACK_BISHOP] |= evalData.attacksFrom[square];
183 break;
184 case WHITE_ROOK:
185 evalData.attacksFrom[square] = getRookAttacks(square, occupiedBB) & ~whitePieces;
186 evalData.attacksByColor[WHITE] |= evalData.attacksFrom[square];
187 evalData.attacksByPiece[WHITE_ROOK] |= evalData.attacksFrom[square];
188 break;
189 case BLACK_ROOK:
190 evalData.attacksFrom[square] = getRookAttacks(square, occupiedBB) & ~blackPieces;
191 evalData.attacksByColor[BLACK] |= evalData.attacksFrom[square];
192 evalData.attacksByPiece[BLACK_ROOK] |= evalData.attacksFrom[square];
193 break;
194 case WHITE_QUEEN:
195 evalData.attacksFrom[square] = queenAttacks(square, occupiedBB) & ~whitePieces;
196 evalData.attacksByColor[WHITE] |= evalData.attacksFrom[square];
197 evalData.attacksByPiece[WHITE_QUEEN] |= evalData.attacksFrom[square];
198 break;
199 case BLACK_QUEEN:
200 evalData.attacksFrom[square] = queenAttacks(square, occupiedBB) & ~blackPieces;
201 evalData.attacksByColor[BLACK] |= evalData.attacksFrom[square];
202 evalData.attacksByPiece[BLACK_QUEEN] |= evalData.attacksFrom[square];
203 break;
204 case WHITE_KING:
205 evalData.attacksFrom[square] = getKingAttacks(square) & ~whitePieces;
206 evalData.attacksByColor[WHITE] |= evalData.attacksFrom[square];
207 evalData.attacksByPiece[WHITE_KING] |= evalData.attacksFrom[square];
208 break;
209 case BLACK_KING:
210 evalData.attacksFrom[square] = getKingAttacks(square) & ~blackPieces;
211 evalData.attacksByColor[BLACK] |= evalData.attacksFrom[square];
212 evalData.attacksByPiece[BLACK_KING] |= evalData.attacksFrom[square];
213 break;
214 default:
215 break;
216 }
217 }

226int getPieceValue(const Piece piece) {
227 switch (piece) {
228 case WHITE_PAWN:
229 case BLACK_PAWN:
230 return 100;
231 case WHITE_KNIGHT:
232 case BLACK_KNIGHT:
233 return 350;
234 case WHITE_BISHOP:
235 case BLACK_BISHOP:
236 return 350;
237 case WHITE_ROOK:
238 case BLACK_ROOK:
239 return 525;
240 case WHITE_QUEEN:
241 case BLACK_QUEEN:
242 return 1000;
243 case WHITE_KING:
244 case BLACK_KING:
245 return 0;
246 default:
247 assert(false);
248 return 0;
249 }
