236 template <Piece piece>
237 void setPiece(const uint8_t square) {
238 assert(piece != Piece::EMPTY);
239 assert(board[square] == Piece::EMPTY);
240 const uint64_t squareBB = squareToBitboard(square);
241
242 board[square] = piece;
243 bitboards[piece] |= squareBB;
244 occupied |= squareBB;
245 colorBoards[getPieceColor(piece)] |= squareBB;
246 pieceCounts[piece] += 1;
247
248 const int zobristIndex = ZOBRIST_PIECE_START_INDEX + static_cast<int>(piece) * SQUARES + square;
249 zobristHash ^= getZobristConstant(zobristIndex);
250 }

352 constexpr PieceColor opponentColor = !color;
353 const uint64_t kingBB = getBitboard<color == WHITE ? WHITE_KING : BLACK_KING>();
