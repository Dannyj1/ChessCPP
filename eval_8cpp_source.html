36int Evaluation::evaluate() {
37 const int phase = calculatePhase();
38 const int modifier = board.getSideToMove() == WHITE ? 1 : -1;
39
40 evaluateMaterial();
41
42 const int whiteScore = ((whiteMidgameScore * (256 - phase)) + (whiteEndgameScore * phase)) / 256;
43 const int blackScore = ((blackMidgameScore * (256 - phase)) + (blackEndgameScore * phase)) / 256;
44
45 return (whiteScore - blackScore) * modifier;
46}

48constexpr int knightPhase = 1;
49constexpr int bishopPhase = 1;
50constexpr int rookPhase = 2;
51constexpr int queenPhase = 4;
52constexpr int totalPhase = knightPhase * 4 + bishopPhase * 4 + rookPhase * 4 + queenPhase * 2;

63int Evaluation::calculatePhase() const {
64 int phase = totalPhase;
65
66 phase -= (board.getPieceCount<WHITE_KNIGHT>() + board.getPieceCount<BLACK_KNIGHT>()) * knightPhase;
67 phase -= (board.getPieceCount<WHITE_BISHOP>() + board.getPieceCount<BLACK_BISHOP>()) * bishopPhase;
68 phase -= (board.getPieceCount<WHITE_ROOK>() + board.getPieceCount<BLACK_ROOK>()) * rookPhase;
69 phase -= (board.getPieceCount<WHITE_QUEEN>() + board.getPieceCount<BLACK_QUEEN>()) * queenPhase;
70
71 return (phase * 256 + (totalPhase / 2)) / totalPhase;
72}

80void Evaluation::evaluateMaterial() {

82 const Piece piece = static_cast<Piece>(pieceIndex);
