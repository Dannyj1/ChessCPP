98template <Direction direction>
99constexpr uint64_t shift(const uint64_t bb) {
100 switch (direction) {
101 case NORTH:
102 return shiftNorth(bb);
103 case SOUTH:
104 return shiftSouth(bb);
105 case EAST:
106 return shiftEast(bb);
107 case WEST:
108 return shiftWest(bb);
109 case NORTH_EAST:
110 return shiftNorthEast(bb);
111 case NORTH_WEST:
112 return shiftNorthWest(bb);
113 case SOUTH_EAST:
114 return shiftSouthEast(bb);
115 case SOUTH_WEST:
116 return shiftSouthWest(bb);
117 case NORTH_NORTH_EAST:
118 return shiftNorthNorthEast(bb);
119 case NORTH_EAST_EAST:
120 return shiftNorthEastEast(bb);
121 case SOUTH_EAST_EAST:
122 return shiftSouthEastEast(bb);
123 case SOUTH_SOUTH_EAST:
124 return shiftSouthSouthEast(bb);
125 case SOUTH_SOUTH_WEST:
126 return shiftSouthSouthWest(bb);
127 case SOUTH_WEST_WEST:
128 return shiftSouthWestWest(bb);
129 default:
130 assert(false);
131 return bb;
132 }
133}

218inline uint64_t calculateKingAttacks(uint64_t bb) {
219 const uint64_t attacks = shiftEast(bb) | shiftWest(bb);
220 bb |= attacks;
221
222 return attacks | shiftNorth(bb) | shiftSouth(bb);
