72std::string getMoveNotation(const Move move) {
73 const uint8_t fromSquare = getFromSquare(move);
74 const uint8_t toSquare = getToSquare(move);
75
76 if (const MoveType moveType = getMoveType(move);

78 const PromotionPiece promotionPiece = getPromotionPiece(move);

91Move getMoveFromMoveNotation(const std::string_view notation) {
92 const Square fromSquare = getSquareFromNotation(notation.substr(0, 2));
93 const Square toSquare = getSquareFromNotation(notation.substr(2, 2));
94
95 // If there is a 5th character, it is a promotion
96 if (notation.size() == 5) {
97 const char promotionPieceChar = notation[4];
98 PromotionPiece promotionPiece;
99
100 switch (promotionPieceChar) {
101 case 'q':
102 promotionPiece = QUEEN_PROMOTION;
103 break;
104 case 'r':
105 promotionPiece = ROOK_PROMOTION;
106 break;
107 case 'b':
108 promotionPiece = BISHOP_PROMOTION;
109 break;
110 case 'n':
111 promotionPiece = KNIGHT_PROMOTION;
112 break;
113 default:
114 assert(false);
115 return 0;
116 }
117
118 return encodeMove(fromSquare, toSquare, promotionPiece);
119 }
120
121 return encodeMove(fromSquare, toSquare);
122}
