26void TranspositionTable::savePosition(const uint64_t zobristHash, const int8_t depth, const int ply, int score,
27 const Move bestMove, const TTNodeType nodeType) const {
28 const uint64_t index = zobristHash & hashSize;
29 TTEntry* entry = &transpositionTable[index];
30
31 // Replace entries only when either:
32 // 1. The new entry is at least as deep as the old entry
33 // 2. The new entry is of the same depth but is from a PV node, while the old entry is not
34 if (depth > entry->depth || (depth == entry->depth && entry->nodeType != EXACT && nodeType == EXACT)) {
35 if (score >= (MATE_SCORE - MAX_PLY)) {
36 score += ply;
37 } else if (score <= (-MATE_SCORE + MAX_PLY)) {
38 score -= ply;
39 }
40
41 score = std::clamp(score, INT16_MIN, INT16_MAX);
42
43 entry->validationHash = zobristHash >> 32;
44 entry->depth = depth;
45 entry->bestMove = bestMove;
46 entry->score = score;
47 entry->nodeType = nodeType;
48 }
49}

51int16_t TranspositionTable::probePosition(const uint64_t zobristHash, const int8_t depth, const int alpha,
52 const int beta, const int ply) const {
53 const uint64_t index = zobristHash & hashSize;
54 const uint32_t validationHash = zobristHash >> 32;
55 const TTEntry* entry = &transpositionTable[index];
56
57 if (entry->validationHash == validationHash && entry->depth >= depth) {
58 bool returnScore = false;
59
60 if (entry->nodeType == EXACT) {
61 returnScore = true;
62 } else if (entry->nodeType == ALPHA) {
63 if (entry->score <= alpha) {
64 returnScore = true;
65 }
66 } else if (entry->nodeType == BETA) {
67 if (entry->score >= beta) {
68 returnScore = true;
69 }
70 }
71
72 if (returnScore) {
73 int adjustedScore = entry->score;
74
75 if (adjustedScore >= MATE_SCORE) {
76 adjustedScore -= ply;
77 } else if (adjustedScore <= -MATE_SCORE) {
78 adjustedScore += ply;
79 }
80
81 return adjustedScore;
82 }
83 }
84
85 return NO_TT_SCORE;
86}

88TTEntry* TranspositionTable::getEntry(const uint64_t zobristHash) const {
89 const uint64_t index = zobristHash & hashSize;
90 TTEntry* entry = &transpositionTable[index];
91
92 // Check validation hash to avoid hash collisions
93 if (entry->validationHash == (zobristHash >> 32)) {
94 return entry;
95 }
96
97 return nullptr;
98}

100void TranspositionTable::setTableSize(int megaBytes) {
101 if ((megaBytes & (megaBytes - 1)) != 0) {
102 megaBytes = 1 << static_cast<int>(log2(megaBytes));
103 }
104
105 const uint64_t byteSize = megaBytes * 1024 * 1024;
106 const uint64_t entryCount = byteSize / sizeof(TTEntry);
107
108 delete[] transpositionTable;
109 transpositionTable = new TTEntry[entryCount]{};
110 hashSize = entryCount - 1;
111
112 std::fill_n(transpositionTable, entryCount, TTEntry{});
113}
