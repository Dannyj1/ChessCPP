36template <PieceColor color, NodeType nodeType>
37int pvSearch(Board& board, int alpha, int beta, int depth) {
38 if (board.isDraw()) {
39 return DRAW_SCORE;
40 }
41
42 if (depth == 0) {
43 return Evaluation(board).evaluate();
44 }
45
46 constexpr bool isPV = nodeType == PV || nodeType == ROOT;
47 constexpr PieceColor opponentColor = !color;
48 int bestScore = std::numeric_limits<int>::min();
49 bool firstMove = true;
50 int legalMoves = 0;
51
52 Move move;
53 MoveList moves = MoveList{};
54 generateMoves<color, ALL>(board, moves);
55 MovePicker movePicker{moves};
56
57 while (movePicker.next(move)) {
58 int score;
59 board.makeMove(move);
60
61 if (!board.isPositionLegal<color>()) {
62 board.unmakeMove();
63 continue;
64 }
65
66 legalMoves += 1;
67
68 if (firstMove) {
69 score = -pvSearch<opponentColor, nodeType>(board, -beta, -alpha, depth - 1);
70 firstMove = false;
71 } else {
72 score = -pvSearch<opponentColor, REGULAR>(board, -alpha - 1, -alpha, depth - 1);
73
74 if (score > alpha && isPV) {
75 score = -pvSearch<opponentColor, PV>(board, -beta, -alpha, depth - 1);
76 }
77 }
78
79 board.unmakeMove();
80
81 if (score >= beta) {
82 return score;
83 }
84
85 if (score > bestScore) {
86 bestScore = score;
87
88 if (score > alpha) {
89 alpha = score;
90 }
91 }
92 }
93
94 if (!legalMoves) {
95 bool inCheck = board.isKingInCheck<color>();
96
97 if (inCheck) {
98 alpha = -MATE_SCORE + board.getPly();
99 } else {
100 alpha = DRAW_SCORE;
101 }
102 }
103
104 return alpha;
105}
