187template <Direction direction>
188constexpr uint64_t shift(const uint64_t bb) {
189 switch (direction) {
190 case NORTH:
191 return shiftNorth(bb);
192 case SOUTH:
193 return shiftSouth(bb);
194 case EAST:
195 return shiftEast(bb);
196 case WEST:
197 return shiftWest(bb);
198 case NORTH_EAST:
199 return shiftNorthEast(bb);
200 case NORTH_WEST:
201 return shiftNorthWest(bb);
202 case SOUTH_EAST:
203 return shiftSouthEast(bb);
204 case SOUTH_WEST:
205 return shiftSouthWest(bb);
206 case NORTH_NORTH_EAST:
207 return shiftNorthNorthEast(bb);
208 case NORTH_EAST_EAST:
209 return shiftNorthEastEast(bb);
210 case SOUTH_EAST_EAST:
211 return shiftSouthEastEast(bb);
212 case SOUTH_SOUTH_EAST:
213 return shiftSouthSouthEast(bb);
214 case SOUTH_SOUTH_WEST:
215 return shiftSouthSouthWest(bb);
216 case SOUTH_WEST_WEST:
217 return shiftSouthWestWest(bb);
218 default:
219 assert(false);
220 return bb;
221 }
222}

425inline uint64_t calculateKingAttacks(uint64_t bb) {
426 const uint64_t attacks = shiftEast(bb) | shiftWest(bb);
427 bb |= attacks;
428 return attacks | shiftNorth(bb) | shiftSouth(bb);
