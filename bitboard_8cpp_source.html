37void initializeAttackLookupTables() {
38 for (uint8_t square = 0; square < SQUARES; ++square) {
39 const uint64_t bb = squareToBitboard(square);
40
41 pawnAttacksTable[WHITE][square] = calculateWhitePawnAttacks(bb);
42 pawnAttacksTable[BLACK][square] = calculateBlackPawnAttacks(bb);
43 knightAttacksTable[square] = calculateKnightAttacks(bb);
44 kingAttacksTable[square] = calculateKingAttacks(bb);
45 }
46}

48void initializeBetweenLookupTable() {
49 for (int from = 0; from < 64; from++) {
50 for (int to = 0; to < 64; to++) {
51 uint64_t m1 = -1ULL;
52 uint64_t a2a7 = 0x0001010101010100ULL;
53 uint64_t b2g7 = 0x0040201008040200ULL;
54 uint64_t h1b7 = 0x0002040810204080ULL;
55 uint64_t btwn, line, rank, file;
56
57 btwn = m1 << from ^ m1 << to;
58 file = (to & 7) - (from & 7);
59 rank = ((to | 7) - from) >> 3;
60 line = (file & 7) - 1 & a2a7;
61 line += 2 * (((rank & 7) - 1) >> 58);
62 line += (rank - file & 15) - 1 & b2g7;
63 line += (rank + file & 15) - 1 & h1b7;
64 line *= btwn & -btwn;
65
66 betweenLookupTable[from][to] = line & btwn;
67 }
68 }
69}

113uint64_t getBishopAttacks(const uint8_t square, uint64_t occupied) {
114 assert(square < SQUARES);
115 occupied &= getBishopMask(square);
116 occupied *= getBishopMagic(square);
117 occupied >>= 64 - BBits[square];
118
119 return getBishopMagicAttacks(square, occupied);
120}

128uint64_t getRookAttacks(const uint8_t square, uint64_t occupied) {
129 assert(square < SQUARES);
130 occupied &= getRookMask(square);
131 occupied *= getRookMagic(square);
132 occupied >>= 64 - RBits[square];
133

143uint64_t queenAttacks(const uint8_t square, const uint64_t occupied) {
144 assert(square < SQUARES);
145 return getBishopAttacks(square, occupied) | getRookAttacks(square, occupied);
