254 template <Piece piece>
255 void setPiece(const uint8_t square) {
256 assert(piece != Piece::EMPTY);
257 assert(board[square] == Piece::EMPTY);
258 const uint64_t squareBB = squareToBitboard(square);
259
260 board[square] = piece;
261 bitboards[piece] |= squareBB;
262 occupied |= squareBB;
263 colorBoards[getPieceColor(piece)] |= squareBB;
264 pieceCounts[piece] += 1;
265
266 const int zobristIndex = ZOBRIST_PIECE_START_INDEX + static_cast<int>(piece) * SQUARES + square;
267 zobristHash ^= getZobristConstant(zobristIndex);
268 }

368 template <PieceColor color>
369 [[nodiscard]] bool isKingInCheck() const {
370 constexpr PieceColor opponentColor = !color;
371 const uint64_t kingBB = getPieceBoard<color == WHITE ? WHITE_KING : BLACK_KING>();
372 const Square kingSquare = bitboardToSquare(kingBB);
373
374 return getSquareAttackersByColor<opponentColor>(kingSquare) != 0;
375 }
