128void benchmark(bool fast) {
129 Engine engine;
130 uint64_t nodes = 0;
131 double totalMs = 0;
132 Board board{};
133
134 engine.registerOptions();
135 engine.doSetup();
136 TranspositionTable::getTT()->setTableSize(128);
137 std::vector<std::string> positions = fast ? FAST_BENCHMARK_POSITIONS : BENCHMARK_POSITIONS;
138 SearchParams params{};
139
140 params.depth = fast ? 3 : 6;
141
142 for (const std::string& position : positions) {
143 for (int i = 0; i < 2; i++) {
144 TranspositionTable::getTT()->reset();
145 const PieceColor color = i == 0 ? WHITE : BLACK;
146
147 board.setFromFEN(position);
148 board.setSideToMove(color);
149 SearchStats stats{};
150 auto start = std::chrono::steady_clock::now();
151
152 if (color == WHITE) {
153 search<WHITE>(engine, board, params, stats);
154 } else {
155 search<BLACK>(engine, board, params, stats);
156 }
157
158 auto end = std::chrono::steady_clock::now();
159 std::chrono::duration<double, std::milli> elapsed = end - start;
160
161 nodes += stats.nodesSearched + stats.qNodesSearched;
162 totalMs += elapsed.count();
163 }
164 }
165
166 if (nodes == 0 || totalMs == 0) {
167 engine.sendMessage("0 nodes 0 nps");
168 return;
169 }
170
171 const double secondsSpent = totalMs / 1000.0;
172 const uint64_t nodesPerSecond = static_cast<uint64_t>(static_cast<double>(nodes) / secondsSpent);
173 std::string message = std::to_string(nodes) + " nodes " + std::to_string(nodesPerSecond) + " nps";
174
175 engine.sendMessage(message);
176}
