233 template <Piece piece>
234 void setPiece(const uint8_t square) {
235 assert(piece != Piece::EMPTY);
236 assert(board[square] == Piece::EMPTY);
237 const uint64_t squareBB = squareToBitboard(square);
238
239 board[square] = piece;
240 bitboards[piece] |= squareBB;
241 occupied |= squareBB;
242 colorBoards[getPieceColor(piece)] |= squareBB;
243
244 const int zobristIndex = ZOBRIST_PIECE_START_INDEX + static_cast<int>(piece) * SQUARES + square;
245 zobristHash ^= getZobristConstant(zobristIndex);
246 }
