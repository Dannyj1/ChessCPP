216 template <Piece piece>
217 void setPiece(const uint8_t square) {
218 assert(piece != Piece::EMPTY);
219 assert(board[square] == Piece::EMPTY);
220 const uint64_t squareBB = squareToBitboard(square);
221
222 board[square] = piece;
223 bitboards[piece] |= squareBB;
224 occupied |= squareBB;
225 colorBoards[getPieceColor(piece)] |= squareBB;
226 pieceCounts[piece] += 1;
227
228 const int zobristIndex = ZOBRIST_PIECE_START_INDEX + static_cast<int>(piece) * SQUARES + square;
229 zobristHash ^= getZobristConstant(zobristIndex);
230 }
