30template <PieceColor movedColor>
31bool Board::isPositionLegal() const {
32 constexpr PieceColor opponentColor = !movedColor;
33 constexpr Piece king = movedColor == WHITE ? WHITE_KING : BLACK_KING;
34 const uint64_t kingBB = getBitboard<king>();
35 const Square kingSquare = bitboardToSquare(kingBB);
36 const Move lastMove = getLastMove();
37 const MoveType lastMoveType = getMoveType(lastMove);
38
39 if (lastMoveType == CASTLING) {
40 const uint8_t fromSquare = getFromSquare(lastMove);
41 const uint8_t toSquare = getToSquare(lastMove);
42 const uint64_t fromSquareAttacks = getSquareAttackersByColor<opponentColor>(fromSquare);
43
44 if (fromSquareAttacks) {
45 // King was in check before castling
46 return false;
47 }
48
49 uint64_t castlingPath = 0;
50
51 if (toSquare == G1) {
52 castlingPath = WHITE_KINGSIDE_CASTLE_PATH;
53 } else if (toSquare == C1) {
54 castlingPath = WHITE_QUEENSIDE_CASTLE_PATH;
55 } else if (toSquare == G8) {
56 castlingPath = BLACK_KINGSIDE_CASTLE_PATH;
57 } else if (toSquare == C8) {
58 castlingPath = BLACK_QUEENSIDE_CASTLE_PATH;
59 }
60
61 while (castlingPath) {
62 const uint8_t square = popLsb(castlingPath);
63 const uint64_t attackers = getSquareAttackersByColor<opponentColor>(square);
64
65 if (attackers) {
66 return false;
67 }
68 }
69 }
70
71 return !getSquareAttackersByColor<opponentColor>(kingSquare);
72}

78template <CastlingRights side>
79bool Board::canCastle() const {
80 assert(side != WHITE_CASTLING && side != BLACK_CASTLING);
81
82 if (!(castlingRights & side)) {
83 return false;
84 }
85
86 uint64_t castlingPath = 0;
87
88 if constexpr (side == WHITE_KINGSIDE) {
89 castlingPath = WHITE_KINGSIDE_CASTLE_UNOCCUPIED;
90 } else if constexpr (side == WHITE_QUEENSIDE) {
91 castlingPath = WHITE_QUEENSIDE_CASTLE_UNOCCUPIED;
92 } else if constexpr (side == BLACK_KINGSIDE) {
93 castlingPath = BLACK_KINGSIDE_CASTLE_UNOCCUPIED;
94 } else if constexpr (side == BLACK_QUEENSIDE) {
95 castlingPath = BLACK_QUEENSIDE_CASTLE_UNOCCUPIED;
96 }
97
98 if (occupied & castlingPath) {
99 return false;
100 }
101
102 return true;
103}

110uint64_t Board::getSquareAttackers(const uint8_t square) const {
111 assert(square < SQUARES);
112 const uint64_t knights = getBitboard<WHITE_KNIGHT>() | getBitboard<BLACK_KNIGHT>();
113 const uint64_t kings = getBitboard<WHITE_KING>() | getBitboard<BLACK_KING>();
114 uint64_t bishopsQueens = getBitboard<WHITE_QUEEN>() | getBitboard<BLACK_QUEEN>();
115 uint64_t rooksQueens = getBitboard<WHITE_QUEEN>() | getBitboard<BLACK_QUEEN>();
116 rooksQueens |= getBitboard<WHITE_ROOK>() | getBitboard<BLACK_ROOK>();
117 bishopsQueens |= getBitboard<WHITE_BISHOP>() | getBitboard<BLACK_BISHOP>();
118
119 return (getPawnAttacks<WHITE>(square) & getBitboard<BLACK_PAWN>())
120 | (getPawnAttacks<BLACK>(square) & getBitboard<WHITE_PAWN>())
121 | (getKnightAttacks(square) & knights)
122 | (getKingAttacks(square) & kings)
123 | (getBishopAttacks(square, occupied) & bishopsQueens)
124 | (getRookAttacks(square, occupied) & rooksQueens);
125}

158void Board::makeMove(const Move& move) {
159 const uint8_t fromSquare = getFromSquare(move);
160 const uint8_t toSquare = getToSquare(move);
161 const MoveType moveType = getMoveType(move);
162 Piece movedPiece = getPieceOnSquare(fromSquare);
163 const PieceType movedPieceType = getPieceType(movedPiece);
164 const Piece capturedPiece = getPieceOnSquare(toSquare);
165
166 history[ply].move = move;
167 history[ply].capturedPiece = capturedPiece;
168 history[ply].enPassantSquare = enPassantSquare;
169 enPassantSquare = 0;
170 history[ply].castlingRights = castlingRights;
171
172 if (capturedPiece != EMPTY) {
173 removePiece(capturedPiece, toSquare);
174
175 if (capturedPiece == WHITE_ROOK) {
176 if (toSquare == A1) {
177 castlingRights &= ~WHITE_QUEENSIDE;
178 } else if (toSquare == H1) {
179 castlingRights &= ~WHITE_KINGSIDE;
180 }
181 } else if (capturedPiece == BLACK_ROOK) {
182 if (toSquare == A8) {
183 castlingRights &= ~BLACK_QUEENSIDE;
184 } else if (toSquare == H8) {
185 castlingRights &= ~BLACK_KINGSIDE;
186 }
187 }
188 }
189
190 removePiece(movedPiece, fromSquare);
191
192 if (moveType == PROMOTION) {
193 const PieceColor color = getPieceColor(movedPiece);
194 const PromotionPiece promotionPieceType = getPromotionPiece(move);
195 const Piece promotionPiece = getPieceFromPromotionPiece(promotionPieceType, color);
196
197 setPiece(promotionPiece, toSquare);
198 } else {
199 setPiece(movedPiece, toSquare);
200 }
201
202 if (moveType == EN_PASSANT) {
203 if (sideToMove == WHITE) {
204 removePiece(BLACK_PAWN, toSquare + SOUTH);
205 } else {
206 removePiece(WHITE_PAWN, toSquare + NORTH);
207 }
208 } else if (moveType == CASTLING) {
209 if (toSquare == G1) {
210 removePiece(WHITE_ROOK, H1);
211 setPiece(WHITE_ROOK, F1);
212 } else if (toSquare == C1) {
213 removePiece(WHITE_ROOK, A1);
214 setPiece(WHITE_ROOK, D1);
215 } else if (toSquare == G8) {
216 removePiece(BLACK_ROOK, H8);
217 setPiece(BLACK_ROOK, F8);
218 } else if (toSquare == C8) {
219 removePiece(BLACK_ROOK, A8);
220 setPiece(BLACK_ROOK, D8);
221 }
222
223 if (sideToMove == WHITE) {
224 castlingRights &= ~WHITE_CASTLING;
225 } else {
226 castlingRights &= ~BLACK_CASTLING;
227 }
228 }
229
230 if (movedPiece == WHITE_KING) {
231 castlingRights &= ~WHITE_CASTLING;
232 } else if (movedPiece == BLACK_KING) {
233 castlingRights &= ~BLACK_CASTLING;
234 } else if (movedPiece == WHITE_ROOK) {
235 if (fromSquare == A1) {
236 castlingRights &= ~WHITE_QUEENSIDE;
237 } else if (fromSquare == H1) {
238 castlingRights &= ~WHITE_KINGSIDE;
239 }
240 } else if (movedPiece == BLACK_ROOK) {
241 if (fromSquare == A8) {
242 castlingRights &= ~BLACK_QUEENSIDE;
243 } else if (fromSquare == H8) {
244 castlingRights &= ~BLACK_KINGSIDE;
245 }
246 }
247
248 if (movedPieceType == PAWN) {
249 if ((fromSquare ^ toSquare) == 16) {
250 if (sideToMove == WHITE) {
251 enPassantSquare = toSquare + SOUTH;
252 } else {
253 enPassantSquare = toSquare + NORTH;
254 }
255 }
256 }
257
258 sideToMove = !sideToMove;
259 assert(ply >= 0 && ply < MAX_PLY);
260 ply++;
261 assert(ply >= 0 && ply < MAX_PLY);
262}

264void Board::unmakeMove() {
265 ply--;
266 assert(ply >= 0 && ply < MAX_PLY);
267 const auto& [move, capturedPiece, enPassantSquare, castlingRights] = history[ply];
268 const uint8_t fromSquare = getFromSquare(move);
269 const uint8_t toSquare = getToSquare(move);
270 const MoveType moveType = getMoveType(move);
271 Piece movedPiece = getPieceOnSquare(toSquare);
272
273 removePiece(movedPiece, toSquare);
274
275 if (moveType == PROMOTION) {
276 const PieceColor color = getPieceColor(movedPiece);
277 movedPiece = static_cast<Piece>(WHITE_PAWN + color);
278 }
279
280 setPiece(movedPiece, fromSquare);
281
282 if (capturedPiece != EMPTY) {
283 setPiece(capturedPiece, toSquare);
284 }
285
286 if (moveType == EN_PASSANT) {
287 const PieceColor movedPieceColor = getPieceColor(movedPiece);
288
289 if (movedPieceColor == WHITE) {
290 setPiece(BLACK_PAWN, toSquare + SOUTH);
291 } else {
292 setPiece(WHITE_PAWN, toSquare + NORTH);
293 }
294 }
295
296 if (moveType == CASTLING) {
297 if (toSquare == G1) {
298 removePiece(WHITE_ROOK, F1);
299 setPiece(WHITE_ROOK, H1);
300 } else if (toSquare == C1) {
301 removePiece(WHITE_ROOK, D1);
302 setPiece(WHITE_ROOK, A1);
303 } else if (toSquare == G8) {
304 removePiece(BLACK_ROOK, F8);
305 setPiece(BLACK_ROOK, H8);
306 } else if (toSquare == C8) {
307 removePiece(BLACK_ROOK, D8);
308 setPiece(BLACK_ROOK, A8);
309 }
310 }
311
312 this->sideToMove = !sideToMove;
313 this->enPassantSquare = enPassantSquare;
314 this->castlingRights = castlingRights;
315}

317void Board::setPieceFromFENChar(const char character, const uint8_t square) {
318 assert(square < SQUARES);
319 // Uppercase char = white, lowercase = black
320 switch (character) {
321 case 'P':
322 setPiece(WHITE_PAWN, square);
323 break;
324 case 'p':
325 setPiece(BLACK_PAWN, square);
326 break;
327 case 'N':
328 setPiece(WHITE_KNIGHT, square);
329 break;
330 case 'n':
331 setPiece(BLACK_KNIGHT, square);
332 break;
333 case 'B':
334 setPiece(WHITE_BISHOP, square);
335 break;
336 case 'b':
337 setPiece(BLACK_BISHOP, square);
338 break;
339 case 'R':
340 setPiece(WHITE_ROOK, square);
341 break;
342 case 'r':
343 setPiece(BLACK_ROOK, square);
344 break;
345 case 'Q':
346 setPiece(WHITE_QUEEN, square);
347 break;
348 case 'q':
349 setPiece(BLACK_QUEEN, square);
350 break;
351 case 'K':
352 setPiece(WHITE_KING, square);
353 break;
354 case 'k':
355 setPiece(BLACK_KING, square);
356 break;
357 default:
358 break;
359 }

362bool Board::setFromFEN(const std::string_view fen) {
363 // TODO: Update zobrist hash once that is implemented
364 uint8_t index = A8;
365 int spaces = 0;
366
367 reset();
368
369 for (const char character : fen) {
370 if (character == ' ') {
371 spaces++;
372 continue;
373 }
374
375 if (character == ',') {
376 break;
377 }
378
379 if (spaces == 0) {
380 if (character == '/') {
381 index -= 16;
382 continue;
383 }
384
385 if (character >= '1' && character <= '8') {
386 index += character - '0';
387 continue;
388 }
389
390 if (character >= 'A' && character <= 'z') {
391 setPieceFromFENChar(character, index);
392 index++;
393 } else {
394 return false;
395 }
396 }
397
398 if (spaces == 1) {
399 if (tolower(character) == 'w') {
400 sideToMove = WHITE;
401 } else if (tolower(character) == 'b') {
402 sideToMove = BLACK;
403 } else {
404 return false;
405 }
406 }
407
408 // TODO: Add zobrist hashes
409 if (spaces == 2) {
410 if (character == '-') {
411 continue;
412 } else if (character == 'K') {
413 castlingRights |= WHITE_KINGSIDE;
414 continue;
415 }
416
417 if (character == 'Q') {
418 castlingRights |= WHITE_QUEENSIDE;
419 continue;
420 }
421
422 if (character == 'k') {
423 castlingRights |= BLACK_KINGSIDE;
424 continue;
425 }
426
427 if (character == 'q') {
428 castlingRights |= BLACK_QUEENSIDE;
429 continue;
430 }
431
432 return false;
433 }
434
435 if (spaces == 3) {
436 if (character == '-') {
437 continue;
438 }
439
440 if (tolower(character) < 'a' || tolower(character) > 'h') {
441 continue;
442 }
443
444 const int8_t file = tolower(character) - 'a';
445 const int8_t rank = (!sideToMove) == WHITE ? 2 : 5;
446
447 if (file < 0 || file > 7) {
448 return false;
449 }
450
451 // TODO: Add enPassant zobrist
452 enPassantSquare = rank * 8 + file;
453 index += 2;
454 }
455
456 // TODO: Implement halfmove and fullmove clock
457 /*if (spaces == 4) {
458 halfMoveClock = character - '0';
459 }
460
461 if (spaces == 5) {
462 fullmoveClock = character - '0';
463 }*/
464 }
465
466 return true;
467}
