41template <PieceColor color, GenerationType type>
42void generateMoves(const Board& board, MoveList& moves) {
43 assert(moves.size == 0);
44
45 // TODO: Implement GenerationType logic using a mask that is computed based on type
46 constexpr Piece opponentKing = color == WHITE ? BLACK_KING : WHITE_KING;
47 const uint64_t ownPieces = board.getColorBitboard<color>();
48 const uint64_t opponentKingBB = board.getBitboard<opponentKing>();
49 const uint64_t genMask = ~(ownPieces | opponentKingBB);
50
51 generatePawnMoves<color, type>(board, moves, genMask);
52 generateKnightMoves<color, type>(board, moves, genMask);
53 generateBishopMoves<color, type>(board, moves, genMask);
54 generateRookMoves<color, type>(board, moves, genMask);
55 generateQueenMoves<color, type>(board, moves, genMask);
56 generateKingMoves<color, type>(board, moves, genMask);
57}

67template <PieceColor color, GenerationType type>
68void generatePawnMoves(const Board& board, MoveList& moves, const uint64_t genMask) {
69 constexpr Piece pawn = color == WHITE ? WHITE_PAWN : BLACK_PAWN;
70 constexpr PieceColor opponentColor = !color;
71
72 const uint64_t pawnBB = board.getBitboard<pawn>();
73 const uint64_t emptyBB = board.getEmptyBitboard();
74 const uint64_t opponentPieces = board.getColorBitboard<opponentColor>();
75 uint64_t pawnSinglePushes;
76 uint64_t pawnDoublePushes;
77 uint64_t pawnWestAttacks;
78 uint64_t pawnEastAttacks;
79
80 if constexpr (color == WHITE) {
81 pawnSinglePushes = whitePawnSinglePush(pawnBB, emptyBB);
82 pawnDoublePushes = whitePawnDoublePush(pawnBB, emptyBB);
83 pawnWestAttacks = whitePawnWestAttacks(pawnBB);
84 pawnEastAttacks = whitePawnEastAttacks(pawnBB);
85 } else {
86 pawnSinglePushes = blackPawnSinglePush(pawnBB, emptyBB);
87 pawnDoublePushes = blackPawnDoublePush(pawnBB, emptyBB);
88 pawnWestAttacks = blackPawnWestAttacks(pawnBB);
89 pawnEastAttacks = blackPawnEastAttacks(pawnBB);
90 }
91
92 uint64_t enPassantMask = squareToBitboard(board.getEnPassantSquare());
93
94 if constexpr (color == WHITE) {
95 enPassantMask &= RANK_6;
96 } else {
97 enPassantMask &= RANK_3;
98 }
99
100 pawnSinglePushes &= genMask;
101 pawnDoublePushes &= genMask;
102 pawnWestAttacks &= (opponentPieces | enPassantMask) & genMask;
103 pawnEastAttacks &= (opponentPieces | enPassantMask) & genMask;
104
105 constexpr Direction fromPushDirection = color == WHITE ? NORTH : SOUTH;
106 constexpr Direction fromSqWestAttackDirection = color == WHITE ? NORTH_WEST : SOUTH_WEST;
107 constexpr Direction fromSqEastAttackDirection = color == WHITE ? NORTH_EAST : SOUTH_EAST;
108 constexpr uint64_t promotionRank = color == WHITE ? RANK_8 : RANK_1;
109
110 while (pawnSinglePushes) {
111 const uint8_t squareTo = popLsb(pawnSinglePushes);
112 const uint64_t squareToBB = squareToBitboard(squareTo);
113 const uint8_t squareFrom = squareTo - fromPushDirection;
114
115 if (squareToBB & promotionRank) {
116 for (const PromotionPiece promotionPiece : {QUEEN_PROMOTION, ROOK_PROMOTION, BISHOP_PROMOTION,
117 KNIGHT_PROMOTION}) {
118 const Move move = encodeMove(squareFrom, squareTo, promotionPiece);
119 moves.moves[moves.size] = move;
120 moves.size++;
121 }
122 } else {
123 const Move move = encodeMove(squareFrom, squareTo);
124
125 moves.moves[moves.size] = move;
126 moves.size++;
127 }
128 }
129
130 while (pawnDoublePushes) {
131 const uint8_t squareTo = popLsb(pawnDoublePushes);
132 const uint8_t squareFrom = squareTo - fromPushDirection - fromPushDirection;
133 const Move move = encodeMove(squareFrom, squareTo);
134
135 moves.moves[moves.size] = move;
136 moves.size++;
137 }
138
139 while (pawnWestAttacks) {
140 const uint8_t squareTo = popLsb(pawnWestAttacks);
141 const uint64_t squareToBB = squareToBitboard(squareTo);
142 const uint8_t squareFrom = squareTo - fromSqWestAttackDirection;
143
144 if (squareTo == board.getEnPassantSquare() && squareToBB & enPassantMask) {
145 const Move move = encodeMove(squareFrom, squareTo, EN_PASSANT);
146 moves.moves[moves.size] = move;
147 moves.size++;
148 } else {
149 if (squareToBB & promotionRank) {
150 for (const PromotionPiece promotionPiece : {QUEEN_PROMOTION, ROOK_PROMOTION, BISHOP_PROMOTION,
151 KNIGHT_PROMOTION}) {
152 const Move move = encodeMove(squareFrom, squareTo, promotionPiece);
153 moves.moves[moves.size] = move;
154 moves.size++;
155 }
156 } else {
157 const Move move = encodeMove(squareFrom, squareTo);
158 moves.moves[moves.size] = move;
159 moves.size++;
160 }
161 }
162 }
163
164 while (pawnEastAttacks) {
165 const uint8_t squareTo = popLsb(pawnEastAttacks);
166 const uint64_t squareToBB = squareToBitboard(squareTo);
167 const uint8_t squareFrom = squareTo - fromSqEastAttackDirection;
168
169 if (squareTo == board.getEnPassantSquare() && squareToBB & enPassantMask) {
170 const Move move = encodeMove(squareFrom, squareTo, EN_PASSANT);
171 moves.moves[moves.size] = move;
172 moves.size++;
173 } else {
174 if (squareToBB & promotionRank) {
175 for (const PromotionPiece promotionPiece : {QUEEN_PROMOTION, ROOK_PROMOTION, BISHOP_PROMOTION,
176 KNIGHT_PROMOTION}) {
177 const Move move = encodeMove(squareFrom, squareTo, promotionPiece);
178 moves.moves[moves.size] = move;
179 moves.size++;
180 }
181 } else {
182 const Move move = encodeMove(squareFrom, squareTo);
183 moves.moves[moves.size] = move;
184 moves.size++;
185 }
186 }
187 }
188}

198template <PieceColor color, GenerationType type>
199void generateKnightMoves(const Board& board, MoveList& moves, const uint64_t genMask) {
200 constexpr Piece knight = color == WHITE ? WHITE_KNIGHT : BLACK_KNIGHT;
201 uint64_t knightBB = board.getBitboard<knight>();
202
203 while (knightBB) {
204 const uint8_t fromSquare = popLsb(knightBB);
205 const uint64_t attacks = getKnightAttacks(fromSquare);
206 uint64_t genBB = attacks & genMask;
207
208 while (genBB) {
209 const uint8_t toSquare = popLsb(genBB);
210 const Move move = encodeMove(fromSquare, toSquare);
211
212 moves.moves[moves.size] = move;
213 moves.size++;
214 }
215 }

310template <PieceColor color, GenerationType type>
311void generateKingMoves(const Board& board, MoveList& moves, const uint64_t genMask) {
312 constexpr Piece king = color == WHITE ? WHITE_KING : BLACK_KING;
313 uint64_t kingBB = board.getBitboard<king>();
314 const uint8_t fromSquare = popLsb(kingBB);
315 uint64_t genBB = getKingAttacks(fromSquare) & genMask;
316
317 while (genBB) {
318 const uint8_t toSquare = popLsb(genBB);
319 const Move move = encodeMove(fromSquare, toSquare);
320
321 moves.moves[moves.size] = move;
322 moves.size++;
323 }
324
325 const uint8_t castlingRights = board.getCastlingRights();
326
327 if constexpr (color == WHITE) {
328 if (castlingRights & WHITE_KINGSIDE && board.canCastle<WHITE_KINGSIDE>()) {
329 const Move move = encodeMove(E1, G1, CASTLING);
330
331 moves.moves[moves.size] = move;
332 moves.size++;
333 }
334
335 if (castlingRights & WHITE_QUEENSIDE && board.canCastle<WHITE_QUEENSIDE>()) {
336 const Move move = encodeMove(E1, C1, CASTLING);
337
338 moves.moves[moves.size] = move;
339 moves.size++;
340 }
341 } else if constexpr (color == BLACK) {
342 if (castlingRights & BLACK_KINGSIDE && board.canCastle<BLACK_KINGSIDE>()) {
343 const Move move = encodeMove(E8, G8, CASTLING);
344
345 moves.moves[moves.size] = move;
346 moves.size++;
347 }
348
349 if (castlingRights & BLACK_QUEENSIDE && board.canCastle<BLACK_QUEENSIDE>()) {
350 const Move move = encodeMove(E8, C8, CASTLING);
351
352 moves.moves[moves.size] = move;
353 moves.size++;
354 }
355 }
