41uint64_t perft(Board& board, const int depth, bool printNodes) {
42 assert(depth >= 0);
43
44 if (depth == 0) {
45 return 1;
46 }
47
48 uint64_t nodes = 0;
49 MoveList moveList{};
50 const PieceColor sideToMove = board.getSideToMove();
51
52 if (sideToMove == WHITE) {
53 generateMoves<WHITE, ALL>(board, moveList);
54 } else {
55 generateMoves<BLACK, ALL>(board, moveList);
56 }
57
58 for (int i = 0; i < moveList.size; i++) {
59 board.makeMove(moveList.moves[i]);
60
61 if (sideToMove == WHITE) {
62 if (!board.isPositionLegal<WHITE>()) {
63 board.unmakeMove();
64 continue;
65 }
66 } else {
67 if (!board.isPositionLegal<BLACK>()) {
68 board.unmakeMove();
69 continue;
70 }
71 }
72
73 uint64_t perftNodes = perft(board, depth - 1, false);
74
75 if (printNodes) {
76 std::string moveNotation = getMoveNotation(moveList.moves[i]);
77
78 std::cout << moveNotation << ": " << perftNodes << std::endl;
79 }
80
81 nodes += perftNodes;
82 board.unmakeMove();
83 }
84
85 return nodes;
86}
